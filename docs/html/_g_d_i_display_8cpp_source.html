66 static int calculateShift(u32 mask)
67 {
68  int m = 0;
69 
70  while(mask) {
71  m++;
72  mask >>= 1;
73  }
74 
75  return m-5;
76 }

96 bool GDIDisplay::initialize()
97 {
98  theApp.sizeX = 240;
99  theApp.sizeY = 160;
100  switch(theApp.videoOption) {
101  case VIDEO_1X:
102  theApp.surfaceSizeX = theApp.sizeX;
103  theApp.surfaceSizeY = theApp.sizeY;
104  break;
105  case VIDEO_2X:
106  theApp.surfaceSizeX = theApp.sizeX * 2;
107  theApp.surfaceSizeY = theApp.sizeY * 2;
108  break;
109  case VIDEO_3X:
110  theApp.surfaceSizeX = theApp.sizeX * 3;
111  theApp.surfaceSizeY = theApp.sizeY * 3;
112  break;
113  case VIDEO_4X:
114  theApp.surfaceSizeX = theApp.sizeX * 4;
115  theApp.surfaceSizeY = theApp.sizeY * 4;
116  break;
117  case VIDEO_320x240:
118  case VIDEO_640x480:
119  case VIDEO_800x600:
120  case VIDEO_OTHER:
121  {
122  int scaleX = (theApp.fsWidth / theApp.sizeX);
123  int scaleY = (theApp.fsHeight / theApp.sizeY);
124  int min = scaleX < scaleY ? scaleX : scaleY;
125  if(theApp.fsMaxScale)
126  min = min > theApp.fsMaxScale ? theApp.fsMaxScale : min;
127  theApp.surfaceSizeX = theApp.sizeX * min;
128  theApp.surfaceSizeY = theApp.sizeY * min;
129  if(theApp.fullScreenStretch) {
130  theApp.surfaceSizeX = theApp.fsWidth;
131  theApp.surfaceSizeY = theApp.fsHeight;
132  }
133  }
134  break;
135  }
136 
137  theApp.rect.left = 0;
138  theApp.rect.top = 0;
139  theApp.rect.right = theApp.sizeX;
140  theApp.rect.bottom = theApp.sizeY;
141 
142  theApp.dest.left = 0;
143  theApp.dest.top = 0;
144  theApp.dest.right = theApp.surfaceSizeX;
145  theApp.dest.bottom = theApp.surfaceSizeY;
146 
147  DWORD style = WS_POPUP | WS_VISIBLE;
148  DWORD styleEx = 0;
149 
150  if(theApp.videoOption <= VIDEO_4X)
151  style |= WS_OVERLAPPEDWINDOW;
152  else
153  styleEx = 0;
154 
155  if(theApp.videoOption <= VIDEO_4X)
156  AdjustWindowRectEx(&theApp.dest, style, TRUE, styleEx);
157  else
158  AdjustWindowRectEx(&theApp.dest, style, FALSE, styleEx);
159 
160  int winSizeX = theApp.dest.right-theApp.dest.left;
161  int winSizeY = theApp.dest.bottom-theApp.dest.top;
162 
163  if(theApp.videoOption > VIDEO_4X) {
164  winSizeX = theApp.fsWidth;
165  winSizeY = theApp.fsHeight;
166  }
167 
168  int x = 0;
169  int y = 0;
170 
171  if(theApp.videoOption <= VIDEO_4X) {
172  x = theApp.windowPositionX;
173  y = theApp.windowPositionY;
174  }
175 
176  // Create a window
177  MainWnd *pWnd = new MainWnd;
178  theApp.m_pMainWnd = pWnd;
179 
180  pWnd->CreateEx(styleEx,
181  theApp.wndClass,
182  "VisualBoyAdvance",
183  style,
184  x,y,winSizeX,winSizeY,
185  NULL,
186  0);
187 
188  if (!(HWND)*pWnd) {
189  winlog("Error creating Window %08x\n", GetLastError());
190  return FALSE;
191  }
192 
193  theApp.updateMenuBar();
194 
195  theApp.adjustDestRect();
196 
197  theApp.mode320Available = false;
198  theApp.mode640Available = false;
199  theApp.mode800Available = false;
200 
201  HDC dc = GetDC(NULL);
202  HBITMAP hbm = CreateCompatibleBitmap(dc, 1, 1);
203  BITMAPINFO *bi = (BITMAPINFO *)info;
204  ZeroMemory(bi, sizeof(info));
205  bi->bmiHeader.biSize = sizeof(BITMAPINFOHEADER);
206  GetDIBits(dc, hbm, 0, 1, NULL, (LPBITMAPINFO)info, DIB_RGB_COLORS);
207  GetDIBits(dc, hbm, 0, 1, NULL, (LPBITMAPINFO)info, DIB_RGB_COLORS);
208  DeleteObject(hbm);
209  ReleaseDC(NULL, dc);
210 
211  if(bi->bmiHeader.biCompression == BI_BITFIELDS) {
212  systemColorDepth = bi->bmiHeader.biBitCount;
213  if(systemColorDepth == 15)
214  systemColorDepth = 16;
215  systemRedShift = calculateShift(*((DWORD *)&bi->bmiColors[0]));
216  systemGreenShift = calculateShift(*((DWORD *)&bi->bmiColors[1]));
217  systemBlueShift = calculateShift(*((DWORD *)&bi->bmiColors[2]));
218  if(systemColorDepth == 16) {
219  if(systemGreenShift == 6) {
220  Init_2xSaI(565);
221  } else {
222  Init_2xSaI(555);
223  }
224  } else if(systemColorDepth == 32)
225  Init_2xSaI(32);
226  } else {
227  systemColorDepth = 32;
228  systemRedShift = 19;
229  systemGreenShift = 11;
230  systemBlueShift = 3;
231 
232  Init_2xSaI(32);
233  }
234  theApp.fsColorDepth = systemColorDepth;
235  if(systemColorDepth == 24)
236  theApp.filterFunction = NULL;
237 #ifdef MMX
238  if(!theApp.disableMMX)
239  cpu_mmx = theApp.detectMMX();
240  else
241  cpu_mmx = 0;
242 #endif
243 
244  utilUpdateSystemColorMaps();
245  theApp.updateFilter();
246  theApp.updateIFB();
247 
248  pWnd->DragAcceptFiles(TRUE);
249 
250  return TRUE;
251 }

267 void GDIDisplay::render()
268 {
269  BITMAPINFO *bi = (BITMAPINFO *)info;
270  bi->bmiHeader.biWidth = theApp.filterWidth+1;
271  bi->bmiHeader.biHeight = -theApp.filterHeight;
272 
273  int pitch = theApp.filterWidth * 2 + 4;
274  if(systemColorDepth == 24)
275  pitch = theApp.filterWidth * 3;
276  else if(systemColorDepth == 32)
277  pitch = theApp.filterWidth * 4 + 4;
278 
279  if(theApp.filterFunction) {
280  bi->bmiHeader.biWidth = theApp.filterWidth * 2;
281  bi->bmiHeader.biHeight = -theApp.filterHeight * 2;
282 
283  if(systemColorDepth == 16)
284  (*theApp.filterFunction)(pix+pitch,
285  pitch,
286  (u8*)theApp.delta,
287  (u8*)filterData,
288  theApp.filterWidth*2*2,
289  theApp.filterWidth,
290  theApp.filterHeight);
291  else
292  (*theApp.filterFunction)(pix+pitch,
293  pitch,
294  (u8*)theApp.delta,
295  (u8*)filterData,
296  theApp.filterWidth*4*2,
297  theApp.filterWidth,
298  theApp.filterHeight);
299  }
300 
301  if(theApp.showSpeed && (theApp.videoOption > VIDEO_4X || theApp.skin != NULL)) {
302  char buffer[30];
303  if(theApp.showSpeed == 1)
304  sprintf(buffer, "%3d%%", systemSpeed);
305  else
306  sprintf(buffer, "%3d%%(%d, %d fps)", systemSpeed,
307  systemFrameSkip,
308  theApp.showRenderedFrames);
309 
310  if(theApp.filterFunction) {
311  int p = theApp.filterWidth * 4;
312  if(systemColorDepth == 24)
313  p = theApp.filterWidth * 6;
314  else if(systemColorDepth == 32)
315  p = theApp.filterWidth * 8;
316  if(theApp.showSpeedTransparent)
317  drawTextTransp((u8*)filterData,
318  p,
319  10,
320  theApp.filterHeight*2-10,
321  buffer);
322  else
323  drawText((u8*)filterData,
324  p,
325  10,
326  theApp.filterHeight*2-10,
327  buffer);
328  } else {
329  if(theApp.showSpeedTransparent)
330  drawTextTransp((u8*)pix,
331  pitch,
332  10,
333  theApp.filterHeight-10,
334  buffer);
335  else
336  drawText((u8*)pix,
337  pitch,
338  10,
339  theApp.filterHeight-10,
340  buffer);
341  }
342  }
343 
344  POINT p;
345  p.x = theApp.dest.left;
346  p.y = theApp.dest.top;
347  CWnd *pWnd = theApp.m_pMainWnd;
348  pWnd->ScreenToClient(&p);
349  POINT p2;
350  p2.x = theApp.dest.right;
351  p2.y = theApp.dest.bottom;
352  pWnd->ScreenToClient(&p2);
353 
354  CDC *dc = pWnd->GetDC();
355 
356  StretchDIBits((HDC)*dc,
357  p.x,
358  p.y,
359  p2.x - p.x,
360  p2.y - p.y,
361  0,
362  0,
363  theApp.rect.right,
364  theApp.rect.bottom,
365  theApp.filterFunction ? filterData : pix+pitch,
366  bi,
367  DIB_RGB_COLORS,
368  SRCCOPY);
369 
370  if(theApp.screenMessage) {
371  if(((GetTickCount() - theApp.screenMessageTime) < 3000) &&
372  !theApp.disableStatusMessage) {
373  dc->SetTextColor(RGB(255,0,0));
374  dc->SetBkMode(TRANSPARENT);
375  dc->TextOut(p.x+10, p2.y - 20, theApp.screenMessageBuffer);
376  } else {
377  theApp.screenMessage = false;
378  }
379  }
380 
381  pWnd->ReleaseDC(dc);
382 }

384 int GDIDisplay::selectFullScreenMode(GUID **)
385 {
386  HWND wnd = GetDesktopWindow();
387  RECT r;
388  GetWindowRect(wnd, &r);
389  int w = (r.right - r.left) & 4095;
390  int h = (r.bottom - r.top) & 4095;
391  HDC dc = GetDC(wnd);
392  int c = GetDeviceCaps(dc, BITSPIXEL);
393  ReleaseDC(wnd, dc);
394 
395  return (c << 24) | (w << 12) | h;
396 }
