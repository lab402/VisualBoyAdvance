137 void gbSoundEvent(register u16 address, register int data)
138 {
139  int freq = 0;
140 
141  gbMemory[address] = data;
142 
143 #ifndef FINAL_VERSION
144  if(soundDebug) {
145  // don't translate. debug only
146  log("Sound event: %08lx %02x\n", address, data);
147  }
148 #endif
149  switch(address) {
150  case NR10:
151  gbMemory[address] = data | 0x80;
152  sound1SweepATL = sound1SweepATLReload = 344 * ((data >> 4) & 7);
153  sound1SweepSteps = data & 7;
154  sound1SweepUpDown = data & 0x08;
155  sound1SweepStep = 0;
156  break;
157  case NR11:
158  gbMemory[address] = data | 0x3f;
159  sound1Wave = soundWavePattern[data >> 6];
160  sound1ATL = sound1ATLreload = 172 * (64 - (data & 0x3f));
161  break;
162  case NR12:
163  sound1EnvelopeVolume = data >> 4;
164  sound1EnvelopeUpDown = data & 0x08;
165  sound1EnvelopeATLReload = sound1EnvelopeATL = 689 * (data & 7);
166  break;
167  case NR13:
168  gbMemory[address] = 0xff;
169  freq1low = data;
170  freq = ((((int)(freq1high & 7)) << 8) | freq1low);
171  sound1ATL = sound1ATLreload;
172  freq = 2048 - freq;
173  if(freq) {
174  sound1Skip = SOUND_MAGIC / freq;
175  } else
176  sound1Skip = 0;
177  break;
178  case NR14:
179  gbMemory[address] = data | 0xbf;
180  freq1high = data;
181  freq = ((((int)(freq1high & 7)) << 8) | freq1low);
182  freq = 2048 - freq;
183  sound1ATL = sound1ATLreload;
184  sound1Continue = data & 0x40;
185  if(freq) {
186  sound1Skip = SOUND_MAGIC / freq;
187  } else
188  sound1Skip = 0;
189  if(data & 0x80) {
190  gbMemory[NR52] |= 1;
191  sound1EnvelopeVolume = gbMemory[NR12] >> 4;
192  sound1EnvelopeUpDown = gbMemory[NR12] & 0x08;
193  sound1EnvelopeATLReload = sound1EnvelopeATL = 689 * (gbMemory[NR12] & 7);
194  sound1SweepATL = sound1SweepATLReload = 344 * ((gbMemory[NR10] >> 4) & 7);
195  sound1SweepSteps = gbMemory[NR10] & 7;
196  sound1SweepUpDown = gbMemory[NR10] & 0x08;
197  sound1SweepStep = 0;
198 
199  sound1Index = 0;
200  sound1On = 1;
201  }
202  break;
203  case NR21:
204  gbMemory[address] = data | 0x3f;
205  sound2Wave = soundWavePattern[data >> 6];
206  sound2ATL = sound2ATLreload = 172 * (64 - (data & 0x3f));
207  break;
208  case NR22:
209  sound2EnvelopeVolume = data >> 4;
210  sound2EnvelopeUpDown = data & 0x08;
211  sound2EnvelopeATLReload = sound2EnvelopeATL = 689 * (data & 7);
212  break;
213  case NR23:
214  gbMemory[address] = 0xff;
215  freq2low = data;
216  freq = (((int)(freq2high & 7)) << 8) | freq2low;
217  sound2ATL = sound2ATLreload;
218  freq = 2048 - freq;
219  if(freq) {
220  sound2Skip = SOUND_MAGIC / freq;
221  } else
222  sound2Skip = 0;
223  break;
224  case NR24:
225  gbMemory[address] = data | 0xbf;
226  freq2high = data;
227  freq = (((int)(freq2high & 7)) << 8) | freq2low;
228  freq = 2048 - freq;
229  sound2ATL = sound2ATLreload;
230  sound2Continue = data & 0x40;
231  if(freq) {
232  sound2Skip = SOUND_MAGIC / freq;
233  } else
234  sound2Skip = 0;
235  if(data & 0x80) {
236  gbMemory[NR52] |= 2;
237  sound2EnvelopeVolume = gbMemory[NR22] >> 4;
238  sound2EnvelopeUpDown = gbMemory[NR22] & 0x08;
239  sound2ATL = sound2ATLreload;
240  sound2EnvelopeATLReload = sound2EnvelopeATL = 689 * (gbMemory[NR22] & 7);
241 
242  sound2Index = 0;
243  sound2On = 1;
244  }
245  break;
246  case NR30:
247  gbMemory[address] = data | 0x7f;
248  if(!(data & 0x80)) {
249  gbMemory[NR52] &= 0xfb;
250  sound3On = 0;
251  }
252  break;
253  case NR31:
254  gbMemory[address] = 0xff;
255  sound3ATL = sound3ATLreload = 172 * (256-data);
256  break;
257  case NR32:
258  gbMemory[address] = data | 0x9f;
259  sound3OutputLevel = (data >> 5) & 3;
260  break;
261  case NR33:
262  gbMemory[address] = 0xff;
263  freq3low = data;
264  freq = 2048 - (((int)(freq3high&7) << 8) | freq3low);
265  if(freq) {
266  sound3Skip = SOUND_MAGIC_2 / freq;
267  } else
268  sound3Skip = 0;
269  break;
270  case NR34:
271  gbMemory[address] = data | 0xbf;
272  freq3high = data;
273  freq = 2048 - (((int)(freq3high&7) << 8) | freq3low);
274  if(freq) {
275  sound3Skip = SOUND_MAGIC_2 / freq;
276  } else {
277  sound3Skip = 0;
278  }
279  sound3Continue = data & 0x40;
280  if((data & 0x80) && (gbMemory[NR30] & 0x80)) {
281  gbMemory[NR52] |= 4;
282  sound3ATL = sound3ATLreload;
283  sound3Index = 0;
284  sound3On = 1;
285  }
286  break;
287  case NR41:
288  sound4ATL = sound4ATLreload = 172 * (64 - (data & 0x3f));
289  break;
290  case NR42:
291  sound4EnvelopeVolume = data >> 4;
292  sound4EnvelopeUpDown = data & 0x08;
293  sound4EnvelopeATLReload = sound4EnvelopeATL = 689 * (data & 7);
294  break;
295  case NR43:
296  freq = freq4 = soundFreqRatio[data & 7];
297  sound4NSteps = data & 0x08;
298 
299  sound4Skip = (freq << 8) / NOISE_MAGIC;
300 
301  sound4Clock = data >> 4;
302 
303  freq = freq / soundShiftClock[sound4Clock];
304 
305  sound4ShiftSkip = (freq << 8) / NOISE_MAGIC;
306 
307  break;
308  case NR44:
309  gbMemory[address] = data | 0xbf;
310  sound4Continue = data & 0x40;
311  if(data & 0x80) {
312  gbMemory[NR52] |= 8;
313  sound4EnvelopeVolume = gbMemory[NR42] >> 4;
314  sound4EnvelopeUpDown = gbMemory[NR42] & 0x08;
315  sound4ATL = sound4ATLreload;
316  sound4EnvelopeATLReload = sound4EnvelopeATL = 689 * (gbMemory[NR42] & 7);
317 
318  sound4On = 1;
319 
320  sound4Index = 0;
321  sound4ShiftIndex = 0;
322 
323  if(sound4NSteps)
324  sound4ShiftRight = 0x7fff;
325  else
326  sound4ShiftRight = 0x7f;
327  }
328  break;
329  case NR50:
330  soundVIN = data & 0x88;
331  soundLevel1 = data & 7;
332  soundLevel2 = (data >> 4) & 7;
333  break;
334  case NR51:
335  soundBalance = (data & soundEnableFlag);
336  break;
337  case NR52:
338  soundMasterOn = data & 0x80;
339  if(!(data & 0x80)) {
340  sound1On = 0;
341  sound2On = 0;
342  sound3On = 0;
343  sound4On = 0;
344  gbMemory[address] &= 0xf0;
345  }
346  gbMemory[address] = data & 0x80 | 0x70 | (gbMemory[address] & 0xf);
347  break;
348  }
349 
350  gbDigitalSound = true;
351 
352  if(sound1On && sound1EnvelopeVolume != 0)
353  gbDigitalSound = false;
354  if(sound2On && sound2EnvelopeVolume != 0)
355  gbDigitalSound = false;
356  if(sound3On && sound3OutputLevel != 0)
357  gbDigitalSound = false;
358  if(sound4On && sound4EnvelopeVolume != 0)
359  gbDigitalSound = false;
360 }

362  void gbSoundChannel1()
363 {
364  int vol = sound1EnvelopeVolume;
365 
366  int freq = 0;
367 
368  int value = 0;
369 
370  if(sound1On && (sound1ATL || !sound1Continue)) {
371  sound1Index += soundQuality*sound1Skip;
372  sound1Index &= 0x1fffffff;
373 
374  value = ((s8)sound1Wave[sound1Index>>24]) * vol;
375  }
376 
377  soundBuffer[0][soundIndex] = value;
378 
379 
380  if(sound1On) {
381  if(sound1ATL) {
382  sound1ATL-=soundQuality;
383 
384  if(sound1ATL <=0 && sound1Continue) {
385  gbMemory[NR52] &= 0xfe;
386  sound1On = 0;
387  }
388  }
389 
390  if(sound1EnvelopeATL) {
391  sound1EnvelopeATL-=soundQuality;
392 
393  if(sound1EnvelopeATL<=0) {
394  if(sound1EnvelopeUpDown) {
395  if(sound1EnvelopeVolume < 15)
396  sound1EnvelopeVolume++;
397  } else {
398  if(sound1EnvelopeVolume)
399  sound1EnvelopeVolume--;
400  }
401 
402  sound1EnvelopeATL += sound1EnvelopeATLReload;
403  }
404  }
405 
406  if(sound1SweepATL) {
407  sound1SweepATL-=soundQuality;
408 
409  if(sound1SweepATL<=0) {
410  freq = (((int)(freq1high & 7)) << 8) | freq1low;
411 
412  int updown = 1;
413 
414  if(sound1SweepUpDown)
415  updown = -1;
416 
417  int newfreq = 0;
418  if(sound1SweepSteps) {
419  newfreq = freq + updown * freq / (1 << sound1SweepSteps);
420  if(newfreq == freq)
421  newfreq = 0;
422  } else
423  newfreq = freq;
424 
425  if(newfreq < 0) {
426  sound1SweepATL += sound1SweepATLReload;
427  } else if(newfreq > 2047) {
428  sound1SweepATL = 0;
429  sound1On = 0;
430  gbMemory[NR52] &= 0xfe;
431  } else {
432  sound1SweepATL += sound1SweepATLReload;
433  sound1Skip = SOUND_MAGIC/(2048 - newfreq);
434 
435  freq1low = newfreq & 0xff;
436  freq1high = (freq1high & 0xf8) |((newfreq >> 8) & 7);
437  }
438  }
439  }
440  }
441 }

537 void gbSoundChannel4()
538 {
539  int vol = sound4EnvelopeVolume;
540 
541  int value = 0;
542 
543  if(sound4Clock <= 0x0c) {
544  if(sound4On && (sound4ATL || !sound4Continue)) {
545  sound4Index += soundQuality*sound4Skip;
546  sound4ShiftIndex += soundQuality*sound4ShiftSkip;
547 
548  if(sound4NSteps) {
549  while(sound4ShiftIndex > 0x1fffff) {
550  sound4ShiftRight = (((sound4ShiftRight << 6) ^
551  (sound4ShiftRight << 5)) & 0x40) |
552  (sound4ShiftRight >> 1);
553  sound4ShiftIndex -= 0x200000;
554  }
555  } else {
556  while(sound4ShiftIndex > 0x1fffff) {
557  sound4ShiftRight = (((sound4ShiftRight << 14) ^
558  (sound4ShiftRight << 13)) & 0x4000) |
559  (sound4ShiftRight >> 1);
560 
561  sound4ShiftIndex -= 0x200000;
562  }
563  }
564 
565  sound4Index &= 0x1fffff;
566  sound4ShiftIndex &= 0x1fffff;
567 
568  value = ((sound4ShiftRight & 1)*2-1) * vol;
569  } else {
570  value = 0;
571  }
572  }
573 
574  soundBuffer[3][soundIndex] = value;
575 
576  if(sound4On) {
577  if(sound4ATL) {
578  sound4ATL-=soundQuality;
579 
580  if(sound4ATL <= 0 && sound4Continue) {
581  gbMemory[NR52] &= 0xf7;
582  sound4On = 0;
583  }
584  }
585 
586  if(sound4EnvelopeATL) {
587  sound4EnvelopeATL-=soundQuality;

603 void gbSoundMix()
604 {
605  int res = 0;
606 
607  if(soundBalance & 16) {
608  res += ((s8)soundBuffer[0][soundIndex]);
609  }
610  if(soundBalance & 32) {
611  res += ((s8)soundBuffer[1][soundIndex]);
612  }
613  if(soundBalance & 64) {
614  res += ((s8)soundBuffer[2][soundIndex]);
615  }
616  if(soundBalance & 128) {
617  res += ((s8)soundBuffer[3][soundIndex]);
618  }
619 
620  if(gbDigitalSound)
621  res *= soundLevel1*256;
622  else
623  res *= soundLevel1*60;
624 
625  if(soundEcho) {
626  res *= 2;
627  res += soundFilter[soundEchoIndex];
628  res /= 2;
629  soundFilter[soundEchoIndex++] = res;
630  }
631 
632  if(soundLowPass) {
633  soundLeft[4] = soundLeft[3];
634  soundLeft[3] = soundLeft[2];
635  soundLeft[2] = soundLeft[1];
636  soundLeft[1] = soundLeft[0];
637  soundLeft[0] = res;
638  res = (soundLeft[4] + 2*soundLeft[3] + 8*soundLeft[2] + 2*soundLeft[1] +
639  soundLeft[0])/14;
640  }
641 
642  switch(soundVolume) {
643  case 0:
644  case 1:
645  case 2:
646  case 3:
647  res *= (soundVolume+1);
648  break;
649  case 4:
650  res >>= 2;
651  break;
652  case 5:
653  res >>= 1;
654  break;
655  }
656 
657  if(res > 32767)
658  res = 32767;
659  if(res < -32768)
660  res = -32768;
661 
662  if(soundReverse)
663  soundFinalWave[++soundBufferIndex] = res;
664  else
665  soundFinalWave[soundBufferIndex++] = res;
666 
667  res = 0;
668 
669  if(soundBalance & 1) {
670  res += ((s8)soundBuffer[0][soundIndex]);
671  }
672  if(soundBalance & 2) {
673  res += ((s8)soundBuffer[1][soundIndex]);
674  }
675  if(soundBalance & 4) {
676  res += ((s8)soundBuffer[2][soundIndex]);
677  }
678  if(soundBalance & 8) {
679  res += ((s8)soundBuffer[3][soundIndex]);
680  }
681 
682  if(gbDigitalSound)
683  res *= soundLevel2*256;
684  else
685  res *= soundLevel2*60;
686 
687  if(soundEcho) {
688  res *= 2;
689  res += soundFilter[soundEchoIndex];
690  res /= 2;
691  soundFilter[soundEchoIndex++] = res;
692 
693  if(soundEchoIndex >= 4000)
694  soundEchoIndex = 0;
695  }
696 
697  if(soundLowPass) {
698  soundRight[4] = soundRight[3];
699  soundRight[3] = soundRight[2];
700  soundRight[2] = soundRight[1];
701  soundRight[1] = soundRight[0];
702  soundRight[0] = res;
703  res = (soundRight[4] + 2*soundRight[3] + 8*soundRight[2] + 2*soundRight[1] +
704  soundRight[0])/14;
705  }
706 
707  switch(soundVolume) {
708  case 0:
709  case 1:
710  case 2:
711  case 3:
712  res *= (soundVolume+1);
713  break;
714  case 4:
715  res >>= 2;
716  break;
717  case 5:
718  res >>= 1;
719  break;
720  }
721 
722  if(res > 32767)
723  res = 32767;
724  if(res < -32768)
725  res = -32768;
726 
727  if(soundReverse)
728  soundFinalWave[-1+soundBufferIndex++] = res;
729  else
730  soundFinalWave[soundBufferIndex++] = res;
731 }

733 void gbSoundTick()
734 {
735  if(systemSoundOn) {
736  if(soundMasterOn) {
737  gbSoundChannel1();
738  gbSoundChannel2();
739  gbSoundChannel3();
740  gbSoundChannel4();
741 
742  gbSoundMix();
743  } else {
744  soundFinalWave[soundBufferIndex++] = 0;
745  soundFinalWave[soundBufferIndex++] = 0;
746  }
747 
748  soundIndex++;
749 
750  if(2*soundBufferIndex >= soundBufferLen) {
751  if(systemSoundOn) {
752  if(soundPaused) {
