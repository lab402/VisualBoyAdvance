438 bool MainWnd::FileRun()
439 {
440  // save battery file before we change the filename...
441  if(rom != NULL || gbRom != NULL) {
442  if(theApp.autoSaveLoadCheatList)
443  winSaveCheatListDefault();
444  writeBatteryFile();
445  cheatSearchCleanup(&cheatSearchData);
446  theApp.emulator.emuCleanUp();
447  remoteCleanUp();
448  emulating = false;
449  }
450  char tempName[2048];
451  char file[2048];
452 
453  utilGetBaseName(theApp.szFile, tempName);
454 
455  _fullpath(file, tempName, 1024);
456  theApp.filename = file;
457 
458  int index = theApp.filename.ReverseFind('.');
459  if(index != -1)
460  theApp.filename = theApp.filename.Left(index);
461 
462  CString ipsname;
463  ipsname.Format("%s.ips", theApp.filename);
464 
465  if(!theApp.dir.GetLength()) {
466  int index = theApp.filename.ReverseFind('\\');
467  if(index != -1) {
468  theApp.dir = theApp.filename.Left(index-1);
469  }
470  }
471 
472  IMAGE_TYPE type = utilFindType(theApp.szFile);
473 
474  if(type == IMAGE_UNKNOWN) {
475  systemMessage(IDS_UNSUPPORTED_FILE_TYPE,
476  "Unsupported file type: %s", theApp.szFile);
477  return false;
478  }
479  systemSaveUpdateCounter = SYSTEM_SAVE_NOT_UPDATED;
480  theApp.cartridgeType = type;
481  if(type == IMAGE_GB) {
482  genericflashcardEnable = theApp.winGenericflashcardEnable;
483 
484 
485  if(!gbLoadRom(theApp.szFile))
486  return false;
487 
488  gbGetHardwareType();
489 
490  // used for the handling of the gb Boot Rom
491  if (gbHardware & 5)
492  {
493  char tempName[2048];
494  GetModuleFileName(NULL, tempName, 2048);
495 
496  char *p = strrchr(tempName, '\\');
497  if(p)
498  *p = 0;
499 
500  strcat(tempName, "\\DMG_ROM.bin");
501 
502  skipBios = theApp.skipBiosFile ? true : false;
503  gbCPUInit(tempName, theApp.useBiosFile ? true : false);
504  }
505 
506 
507 
508  gbReset();
509  theApp.emulator = GBSystem;
510  gbBorderOn = theApp.winGbBorderOn;
511  theApp.romSize = gbRomSize;
512 
513 
514  if(theApp.autoIPS) {
515  int size = gbRomSize;
516  utilApplyIPS(ipsname, &gbRom, &size);
517  if(size != gbRomSize) {
518  extern bool gbUpdateSizes();
519  gbUpdateSizes();
520  gbReset();
521  theApp.romSize = size;
522  }
523  }
524  } else {
525  int size = CPULoadRom(theApp.szFile);
526  if(!size)
527  return false;
528 
529  theApp.romSize = size;
530 
531  flashSetSize(theApp.winFlashSize);
532  rtcEnable(theApp.winRtcEnable);
533  cpuSaveType = theApp.winSaveType;
534 
535  GetModuleFileName(NULL, tempName, 2048);
536 
537  char *p = strrchr(tempName, '\\');
538  if(p)
539  *p = 0;
540 
541  char buffer[5];
542  strncpy(buffer, (const char *)&rom[0xac], 4);
543  buffer[4] = 0;
544 
545  strcat(tempName, "\\vba-over.ini");
546 
547  UINT i = GetPrivateProfileInt(buffer,
548  "rtcEnabled",
549  -1,
550  tempName);
551  if(i != (UINT)-1)
552  rtcEnable(i == 0 ? false : true);
553 
554  i = GetPrivateProfileInt(buffer,
555  "flashSize",
556  -1,
557  tempName);
558  if(i != (UINT)-1 && (i == 0x10000 || i == 0x20000))
559  flashSetSize((int)i);
560 
561  i = GetPrivateProfileInt(buffer,
562  "saveType",
563  -1,
564  tempName);
565  if(i != (UINT)-1 && (i <= 5))
566  cpuSaveType = (int)i;
567  i = GetPrivateProfileInt(buffer,
568  "mirroringEnabled",
569  -1,
570  tempName);
571  if(i != (UINT)-1)
572  doMirroring (i == 0 ? false : true);
573 
574  theApp.emulator = GBASystem;
575  /* disabled due to problems
576  if(theApp.removeIntros && rom != NULL) {
577  *((u32 *)rom)= 0xea00002e;
578  }
579  */
580 
581  if(theApp.autoIPS) {
582  int size = 0x2000000;
583  utilApplyIPS(ipsname, &rom, &size);
584  if(size != 0x2000000) {
585  CPUReset();
586  }
587  }
588  }
589 
590  if(theApp.soundInitialized) {

592  gbSoundReset();
593  else
594  soundReset();
595  } else {

608

611

613 
614  theApp.updateWindowSize(theApp.videoOption);
615

617

620 
621  emulating = true;

625

629

632  theApp.rewindSaveNeeded = false;

637 void MainWnd::OnInitMenuPopup(CMenu* pMenu, UINT nIndex, BOOL bSysMenu)
638 {
639  ASSERT(pMenu != NULL);
640 
641  CCmdUI state;
642  state.m_pMenu = pMenu;
643  ASSERT(state.m_pOther == NULL);
644  ASSERT(state.m_pParentMenu == NULL);
645 
646  // determine if menu is popup in top-level menu and set m_pOther to
647  // it if so (m_pParentMenu == NULL indicates that it is secondary popup)
648  HMENU hParentMenu;
649  if (AfxGetThreadState()->m_hTrackingMenu == pMenu->m_hMenu)
650  state.m_pParentMenu = pMenu; // parent == child for tracking popup
651  else if ((hParentMenu = ::GetMenu(m_hWnd)) != NULL) {
652  CWnd* pParent = GetTopLevelParent();
653  // child windows don't have menus -- need to go to the top!
654  if (pParent != NULL &&
655  (hParentMenu = ::GetMenu(pParent->m_hWnd)) != NULL) {
656  int nIndexMax = ::GetMenuItemCount(hParentMenu);
657  for (int nIndex = 0; nIndex < nIndexMax; nIndex++) {
658  if (::GetSubMenu(hParentMenu, nIndex) == pMenu->m_hMenu) {
659  // when popup is found, m_pParentMenu is containing menu
660  state.m_pParentMenu = CMenu::FromHandle(hParentMenu);
661  break;
662  }
663  }
664  }
665  }
666 
667  state.m_nIndexMax = pMenu->GetMenuItemCount();
668  for (state.m_nIndex = 0; state.m_nIndex < state.m_nIndexMax;
669  state.m_nIndex++) {
670  state.m_nID = pMenu->GetMenuItemID(state.m_nIndex);
671  if (state.m_nID == 0)
672  continue; // menu separator or invalid cmd - ignore it
673 
674  ASSERT(state.m_pOther == NULL);
675  ASSERT(state.m_pMenu != NULL);
676  if (state.m_nID == (UINT)-1) {
677  // possibly a popup menu, route to first item of that popup
678  state.m_pSubMenu = pMenu->GetSubMenu(state.m_nIndex);
679  if (state.m_pSubMenu == NULL ||
680  (state.m_nID = state.m_pSubMenu->GetMenuItemID(0)) == 0 ||
681  state.m_nID == (UINT)-1) {
682  continue; // first item of popup can't be routed to
683  }
684  state.DoUpdate(this, FALSE); // popups are never auto disabled
685  } else {
686  // normal menu item
687  // Auto enable/disable if frame window has 'm_bAutoMenuEnable'
688  // set and command is _not_ a system command.
689  state.m_pSubMenu = NULL;
690  state.DoUpdate(this, state.m_nID < 0xF000);
691  }
692 
693  // adjust for menu deletions and additions
694  UINT nCount = pMenu->GetMenuItemCount();
695  if (nCount < state.m_nIndexMax) {
696  state.m_nIndex -= (state.m_nIndexMax - nCount);
697  while (state.m_nIndex < nCount &&
698  pMenu->GetMenuItemID(state.m_nIndex) == state.m_nID) {
699  state.m_nIndex++;
700  }
701  }
702  state.m_nIndexMax = nCount;
703  }
704 }
