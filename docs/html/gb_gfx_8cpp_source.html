422 void gbDrawSpriteTile(int tile, int x,int y,int t, int flags,
423  int size,int spriteNumber)
424 {
425  u8 * bank0;
426  u8 * bank1;
427  if(gbCgbMode) {
428  if(register_VBK & 1) {
429  bank0 = &gbVram[0x0000];
430  bank1 = &gbVram[0x2000];
431  } else {
432  bank0 = &gbVram[0x0000];
433  bank1 = &gbVram[0x2000];
434  }
435  } else {
436  bank0 = &gbMemory[0x8000];
437  bank1 = NULL;
438  }
439 
440  int init = 0x0000;
441 
442  for (int i = 0; i<4; i++)
443  {
444  gbObp0[i] = (gbObp0Line[x+11+gbSpritesTicks[x]*(gbSpeed ? 2 : 4)]>>(i<<1)) & 3;
445  gbObp1[i] = (gbObp1Line[x+11+gbSpritesTicks[x]*(gbSpeed ? 2 : 4)]>>(i<<1)) & 3;
446  }
447  u8 *pal = gbObp0;
448 
449  int flipx = (flags & 0x20);
450  int flipy = (flags & 0x40);
451 
452  if((flags & 0x10))
453  pal = gbObp1;
454 
455  if(flipy) {
456  t = (size ? 15 : 7) - t;
457  }
458 
459  int prio = flags & 0x80;
460 
461  int address = init + tile * 16 + 2*t;
462  int a = 0;
463  int b = 0;
464 
465  if(gbCgbMode && (flags & 0x08)) {
466  a = bank1[address++];
467  b = bank1[address++];
468  } else {
469  a = bank0[address++];
470  b = bank0[address++];
471  }
472 
473  for(int xx = 0; xx < 8; xx++) {
474  u8 mask = 1 << (7-xx);
475  u8 c = 0;
476  if( (a & mask))
477  c++;
478  if( (b & mask))
479  c+=2;
480 
481  if(c==0) continue;
482 
483  int xxx = xx+x;
484  if(flipx)
485  xxx = (7-xx+x);
486 
487  if(xxx < 0 || xxx > 159)
488  continue;
489 
490  u16 color = gbLineBuffer[xxx];
491 
492  // Fixes OAM-BG priority
493  if(prio && (register_LCDC & 1)) {
494  if(color < 0x200 && ((color & 0xFF) != 0))
495  continue;
496  }
497  // Fixes OAM-BG priority for Moorhuhn 2
498  if(color >= 0x300 && color != 0x300 && (register_LCDC & 1))
499  continue;
500  else if(color >= 0x200 && color < 0x300) {
501  int sprite = color & 0xff;
502 
503  int spriteX = gbMemory[0xfe00 + 4 * sprite + 1] - 8;
504 
505  if(spriteX == x) {
506  if(sprite < spriteNumber)
507  continue;
508  } else {
509  if(gbCgbMode) {
510  if(sprite < spriteNumber)
511  continue;
512  } else {
513  // Fixes GB sprites priorities (was '< x + 8' before)
514  // ('A boy and his blob...' sprites' emulation is now correct)
515  if(spriteX < x)
516  continue;
517  }
518  }
519  }
520 
521 
522  gbLineBuffer[xxx] = 0x200 + spriteNumber;
523 
524  // make sure that sprites will work even in CGB mode
525  if(gbCgbMode) {
526  c = c + (flags & 0x07)*4 + 32;
527  } else {
528  c = pal[c];
529 
530  if(gbSgbMode && !gbCgbMode) {
531  int dx = xxx >> 3;
532  int dy = y >> 3;
533 
534  int palette = gbSgbATF[dy * 20 + dx];
535 
536  if(c == 0)
537  palette = 0;
538 
539  c = c + 4*palette;
540  } else {
541  c += 4;
542  }
543  }
544 
545  gbLineMix[xxx] = gbColorOption ? gbColorFilter[gbPalette[c] & 0x7FFF] :
546  gbPalette[c] & 0x7FFF;
547  }
