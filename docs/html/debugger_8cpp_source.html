273 static void debuggerPrintBaseType(Type *t, u32 value, u32 location,
274  LocationType type,
275  int bitSize, int bitOffset)
276 {
277  if(bitSize) {
278  if(bitOffset)
279  value >>= ((t->size*8)-bitOffset-bitSize);
280  value &= (1 << bitSize)-1;
281  } else {
282  if(t->size == 2)
283  value &= 0xFFFF;
284  else if(t->size == 1)
285  value &= 0xFF;
286  }
287 
288  if(t->size == 8) {
289  u64 value = 0;
290  if(type == LOCATION_memory) {
291  value = debuggerReadMemory(location) |
292  ((u64)debuggerReadMemory(location+4)<<32);
293  } else if(type == LOCATION_register) {
294  value = reg[location].I | ((u64)reg[location+1].I << 32);
295  }
296  switch(t->encoding) {
297  case DW_ATE_signed:
298  switch(debuggerRadix) {
299  case 0:
300  printf("%lld", value);
301  break;
302  case 1:
303  printf("0x%llx", value);
304  break;
305  case 2:
306  printf("0%llo", value);
307  break;
308  }
309  break;
310  case DW_ATE_unsigned:
311  switch(debuggerRadix) {
312  case 0:
313  printf("%llu", value);
314  break;
315  case 1:
316  printf("0x%llx", value);
317  break;
318  case 2:
319  printf("0%llo", value);
320  break;
321  }
322  break;
323  default:
324  printf("Unknowing 64-bit encoding\n");
325  }
326  return;
327  }
328 
329  switch(t->encoding) {
330  case DW_ATE_boolean:
331  if(value)
332  printf("true");
333  else
334  printf("false");
335  break;
336  case DW_ATE_signed:
337  switch(debuggerRadix) {
338  case 0:
339  printf("%d", value);
340  break;
341  case 1:
342  printf("0x%x", value);
343  break;
344  case 2:
345  printf("0%o", value);
346  break;
347  }
348  break;
349  case DW_ATE_unsigned:
350  case DW_ATE_unsigned_char:
351  switch(debuggerRadix) {
352  case 0:
353  printf("%u", value);
354  break;
355  case 1:
356  printf("0x%x", value);
357  break;
358  case 2:
359  printf("0%o", value);
360  break;
361  }
362  break;
363  default:
364  printf("UNKNOWN BASE %d %08x", t->encoding, value);
365  }
366 }

427 static void debuggerPrintMember(Function *f,
428  Member *m,
429  u32 objLocation,
430  u32 location)
431 {
432  int bitSize = m->bitSize;
433  if(bitSize) {
434  u32 value = 0;
435  int off = m->bitOffset;
436  int size = m->byteSize;
437  u32 v = 0;
438  if(size == 1)
439  v = debuggerReadByte(location);
440  else if(size == 2)
441  v = debuggerReadHalfWord(location);
442  else if(size == 4)
443  v = debuggerReadMemory(location);
444 
445  while(bitSize) {
446  int top = size*8 - off;
447  int bot = top - bitSize;
448  top--;
449  if(bot >= 0) {
450  value = (v >> (size*8 - bitSize - off)) & ((1 << bitSize)-1);
451  bitSize = 0;
452  } else {
453  value |= (v & ((1 << top)-1)) << (bitSize - top);
454  bitSize -= (top+1);
455  location -= size;
456  off = 0;
457  if(size == 1)
458  v = debuggerReadByte(location);
459  else if(size == 2)
460  v = debuggerReadHalfWord(location);
461  else
462  v = debuggerReadMemory(location);
463  }
464  }
465  debuggerPrintBaseType(m->type, value, location, LOCATION_memory,
466  bitSize, 0);
467  } else {
468  debuggerPrintValueInternal(f, m->type, m->location, m->bitSize,
469  m->bitOffset, objLocation);
470  }
471 }

1656  else { start = workRAM + (SearchStart & 0x3FFFF); end = workRAM + 0x3FFFF; break; };

1672  case 8:
1673  case 9:
1674  case 10:
1675  case 11:
1676  case 12:
1677  case 13:

1683  return;
