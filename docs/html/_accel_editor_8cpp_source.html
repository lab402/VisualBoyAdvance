136 void AccelEditor::OnSelchangeCommands()

173 void AccelEditor::OnAssign()
174 {
175  // Control if it's not already affected
176  CCmdAccelOb* pCmdAccel;
177  CAccelsOb* pAccel;
178  WORD wIDCommand;
179  POSITION pos;
180 
181  WORD wKey;
182  bool bCtrl, bAlt, bShift;
183 
184  if (!m_key.GetAccelKey(wKey, bCtrl, bAlt, bShift))
185  return; // no valid key, abort
186 
187  int count = m_commands.GetCount();
188  int index;
189  for (index = 0; index < count; index++) {
190 
191  wIDCommand = LOWORD(m_commands.GetItemData(index));
192  mgr.m_mapAccelTable.Lookup(wIDCommand, pCmdAccel);
193 
194  pos = pCmdAccel->m_Accels.GetHeadPosition();
195  while (pos != NULL) {
196  pAccel = pCmdAccel->m_Accels.GetNext(pos);
197  if (pAccel->IsEqual(wKey, bCtrl, bAlt, bShift)) {
198  // the key is already affected (in the same or other command)
199  m_alreadyAffected.SetWindowText(pCmdAccel->m_szCommand);
200  m_key.SetSel(0, -1);
201  return; // abort
202  }
203  }
204  }
205 
206  // OK, we can add the accel key in the currently selected group
207  index = m_commands.GetCurSel();
208  if (index == LB_ERR)
209  return;
210 
211  // Get the object who manage the accels list, associated to the command.
212  wIDCommand = LOWORD(m_commands.GetItemData(index));
213 
214  if (mgr.m_mapAccelTable.Lookup(wIDCommand, pCmdAccel) != TRUE)
215  return;
216 
217  BYTE cVirt = 0;
218  if (bCtrl)
219  cVirt |= FCONTROL;
220  if (bAlt)
221  cVirt |= FALT;
222  if (bShift)
223  cVirt |= FSHIFT;
224 
225  cVirt |= FVIRTKEY;
226 
227  // Create the new key...
228  pAccel = new CAccelsOb(cVirt, wKey, false);
229  ASSERT(pAccel != NULL);
230  // ...and add in the list.
231  pCmdAccel->m_Accels.AddTail(pAccel);
232 
233  // Update the listbox.
234  CString szBuffer;
235  pAccel->GetString(szBuffer);
236 
237  index = m_currents.AddString(szBuffer);
238  m_currents.SetItemData(index, (DWORD_PTR)pAccel);
239 
240  // Reset the key editor.
241  m_key.ResetKey();
242 }

244 void AccelEditor::OnRemove()
245 {
246  // Some controls
247  int indexCurrent = m_currents.GetCurSel();
248  if (indexCurrent == LB_ERR)
249  return;
250 
251  // 2nd part.
252  int indexCmd = m_commands.GetCurSel();
253  if (indexCmd == LB_ERR)
254  return;
255 
256  // Ref to the ID command
257  WORD wIDCommand = LOWORD(m_commands.GetItemData(indexCmd));
258 
259  // Run through the accels,and control if it can be deleted.
260  CCmdAccelOb* pCmdAccel;
261  if (mgr.m_mapAccelTable.Lookup(wIDCommand, pCmdAccel) == TRUE) {
262  CAccelsOb* pAccel;
263  CAccelsOb* pAccelCurrent = (CAccelsOb*)(m_currents.GetItemData(indexCurrent));
264  CString szBuffer;
265  POSITION pos = pCmdAccel->m_Accels.GetHeadPosition();
266  POSITION PrevPos;
267  while (pos != NULL) {
268  PrevPos = pos;
269  pAccel = pCmdAccel->m_Accels.GetNext(pos);
270  if (pAccel == pAccelCurrent) {
271  if (!pAccel->m_bLocked) {
272  // not locked, so we delete the key
273  pCmdAccel->m_Accels.RemoveAt(PrevPos);
274  delete pAccel;
275  // and update the listboxes/key editor/static text
276  m_currents.DeleteString(indexCurrent);
277  m_key.ResetKey();
278  m_alreadyAffected.SetWindowText("");
279  return;
280  } else {
281  systemMessage(0,"Unable to remove this\naccelerator (Locked)");
282  return;
283  }
284  }
285  }
286  systemMessage(0,"internal error (CAccelDlgHelper::Remove : pAccel unavailable)");
287  return;
288  }
289  systemMessage(0,"internal error (CAccelDlgHelper::Remove : Lookup failed)");
290 }
