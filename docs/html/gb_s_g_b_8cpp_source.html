171 void gbSgbRenderScreenToBuffer()
172 {
173  u16 mapAddress = 0x9800;
174 
175  if(register_LCDC & 0x08)
176  mapAddress = 0x9c00;
177 
178  u16 patternAddress = 0x8800;
179 
180  int flag = 1;
181 
182  if(register_LCDC & 0x10) {
183  patternAddress = 0x8000;
184  flag = 0;
185  }
186 
187  u8 *toAddress = gbSgbScreenBuffer;
188 
189  for(int i = 0; i < 13; i++) {
190  for(int j = 0; j < 20; j++) {
191  int tile = getmem(mapAddress);
192  mapAddress++;
193 
194  if(flag) {
195  if(tile > 127)
196  tile -= 128;
197  else
198  tile += 128;
199  }
200  for(int k = 0; k < 16; k++)
201  *toAddress++ = getmem(patternAddress + tile*16 + k);
202  }
203  mapAddress += 12;
204  }
205 }

207 void gbSgbDrawBorderTile(int x, int y, int tile, int attr)
208 {
209  u16 *dest = (u16*)pix + ((y+1) * (256+2)) + x;
210  u8 *dest8 = (u8*)pix + ((y*256)+x)*3;
211  u32 *dest32 = (u32*)pix + ((y+1)*257) + x;
212 
213  u8 *tileAddress = &gbSgbBorderChar[tile * 32];
214  u8 *tileAddress2 = &gbSgbBorderChar[tile * 32 + 16];
215 
216  u8 l = 8;
217 
218  u8 palette = ((attr >> 2 ) & 7);
219 
220  if(palette < 4)
221  palette += 4;
222 
223  palette *= 16;
224 
225  u8 xx = 0;
226  u8 yy = 0;
227 
228  int flipX = attr & 0x40;
229  int flipY = attr & 0x80;
230 
231  while(l > 0) {
232  u8 mask = 0x80;
233  u8 a = *tileAddress++;
234  u8 b = *tileAddress++;
235  u8 c = *tileAddress2++;
236  u8 d = *tileAddress2++;
237 
238  while(mask > 0) {
239 
240  u8 color = 0;
241  if(a & mask)
242  color++;
243  if(b & mask)
244  color+=2;
245  if(c & mask)
246  color+=4;
247  if(d & mask)
248  color+=8;
249 
250  u8 xxx = xx;
251  u8 yyy = yy;
252 
253  if(flipX)
254  xxx = 7 - xx;
255  if(flipY)
256  yyy = 7 - yy;
257 
258  u16 c = gbPalette[palette + color];
259 
260  // Fix for Super Snaky ???
261  // (it allows SGB borders to not redraw on the GB screen)
262  //if(!color)
263  // c = gbPalette[0];
264  if(((yy < 40 || yy >= 184) || (xx < 48 || xx >= 208)) && (color || (gbSgbResetFlag == true))) {
265  switch(systemColorDepth) {
266  case 16:
267  gbSgbDraw16Bit(dest + yyy*(256+2) + xxx, c);
268  break;
269  case 24:
270  gbSgbDraw24Bit(dest8 + (yyy*256+xxx)*3, c);
271  break;
272  case 32:
273  gbSgbDraw32Bit(dest32 + yyy*(256+1)+xxx, c);
274  break;
275  }
276  }
277 
278  mask >>= 1;
279 
280  xx++;
281  }
282  yy++;
283  xx = 0;
284  l--;
285  mask = 0x80;
286  }
287 }

410 void gbSgbAttributeBlock()
411 {
412  u8 *fromAddress = &gbSgbPacket[1];
413 
414  u8 nDataSet = *fromAddress++;
415  if(nDataSet > 12)
416  nDataSet = 12;
417  if(nDataSet == 0)
418  nDataSet = 1;
419 
420  while(nDataSet) {
421  u8 controlCode = (*fromAddress++) & 7;
422  u8 paletteDesignation = (*fromAddress++) & 0x3f;
423  u8 startH = (*fromAddress++) & 0x1f;
424  u8 startV = (*fromAddress++) & 0x1f;
425  u8 endH = (*fromAddress++) & 0x1f;
426  u8 endV = (*fromAddress++) & 0x1f;
427 
428  u8 * toAddress = gbSgbATF;
429 
430  for(u8 y = 0; y < 18; y++) {
431  for(u8 x = 0; x < 20; x++) {
432  if(x < startH || y < startV ||
433  x > endH || y > endV) {
434  // outside
435  if(controlCode & 0x04)
436  *toAddress = (paletteDesignation >> 4) & 0x03;
437  } else if(x > startH && x < endH &&
438  y > startV && y < endV) {
439  // inside
440  if(controlCode & 0x01)
441  *toAddress = paletteDesignation & 0x03;
442  } else {
443  // surrounding line
444  if(controlCode & 0x02)
445  *toAddress = (paletteDesignation>>2) & 0x03;
446  else if(controlCode == 0x01)
447  *toAddress = paletteDesignation & 0x03;
448  }
449  toAddress++;
450  }
451  }
452  nDataSet--;
453  }
454 }

550 void gbSgbAttributeCharacter()
551 {
552  u8 startH = gbSgbPacket[1] & 0x1f;
553  u8 startV = gbSgbPacket[2] & 0x1f;
554  int nDataSet = READ16LE(((u16 *)&gbSgbPacket[3]));
555  int style = gbSgbPacket[5] & 1;
556  if(startH > 19)
557  startH = 19;
558  if(startV > 17)
559  startV = 17;
560 
561  u8 s = 6;
562  u8 *fromAddress = &gbSgbPacket[6];
563  u8 v = *fromAddress++;
564 
565  if(style) {
566  while(nDataSet) {
567  u8 p = (v >> s) & 3;
568  gbSgbATF[startV * 20 + startH] = p;
569  startV++;
570  if(startV == 18) {
571  startV = 0;
572  startH++;
573  if(startH == 20)
574  break;
575  }
576 
577  if(s)
578  s -= 2;
579  else {
580  s = 6;
581  v = *fromAddress++;
582  nDataSet--;
583  }
584  }
585  } else {
586  while(nDataSet) {
587  u8 p = (v >> s) & 3;
588  gbSgbATF[startV * 20 + startH] = p;
589  startH++;
590  if(startH == 20) {
591  startH = 0;
592  startV++;
593  if(startV == 18)
594  break;
595  }
596 
597  if(s)
598  s -= 2;
599  else {
600  s = 6;
601  v = *fromAddress++;
602  nDataSet--;
603  }
604  }
605  }

701 void gbSgbCommand()
702 {
703  int command = gbSgbPacket[0] >> 3;
704  // int nPacket = gbSgbPacket[0] & 7;
705 
706  switch(command) {
707  case 0x00:
708  gbSgbSetPalette(0,1,(u16 *)&gbSgbPacket[1]);
709  break;
710  case 0x01:
711  gbSgbSetPalette(2,3,(u16 *)&gbSgbPacket[1]);
712  break;
713  case 0x02:
714  gbSgbSetPalette(0,3,(u16 *)&gbSgbPacket[1]);
715  break;
716  case 0x03:
717  gbSgbSetPalette(1,2,(u16 *)&gbSgbPacket[1]);
718  break;
719  case 0x04:
720  gbSgbAttributeBlock();
721  break;
722  case 0x05:
723  gbSgbAttributeLine();
724  break;
725  case 0x06:
726  gbSgbAttributeDivide();
727  break;
728  case 0x07:
729  gbSgbAttributeCharacter();
730  break;
731  case 0x0a:
732  gbSgbSetPalette();
733  break;
734  case 0x0b:
735  gbSgbScpPalette();
736  break;
737  case 0x11:

764 void gbSgbDoBitTransfer(u8 value)
765 {
766  value = value & 0x30;
767  switch(gbSgbPacketState) {
768  case GBSGB_NONE:
769  if(value == 0) {
770  gbSgbPacketState = GBSGB_RESET;
771  gbSgbPacketTimeout = GBSGB_PACKET_TIMEOUT;
772  } else if (value == 0x30) {
773  if(gbSgbMultiplayer) {
774  if((gbSgbReadingController & 7) == 7) {
775  gbSgbReadingController = 0;
776  if(gbSgbMultiplayer) {
777  gbSgbNextController--;
778  if(gbSgbFourPlayers) {
779  if(gbSgbNextController == 0x0b)
780  gbSgbNextController = 0x0f;
781  } else {
782  if(gbSgbNextController == 0x0d)
783  gbSgbNextController = 0x0f;
784  }
785  }
786  } else {
787  gbSgbReadingController &= 3;
788  }
789  }
790  gbSgbPacketTimeout = 0;
791  } else {
792  if(value == 0x10)
793  gbSgbReadingController |= 0x2;
794  else if(value == 0x20)
795  gbSgbReadingController |= 0x01;
796  gbSgbPacketTimeout = 0;
797  }
798  gbSgbPacketTimeout = 0;
799  break;
800  case GBSGB_RESET:
801  if(value == 0x30) {
802  gbSgbPacketState = GBSGB_PACKET_TRANSMIT;
803  gbSgbPacketByte = 0;
804  gbSgbPacketNBits = 0;
805  gbSgbPacketTimeout = GBSGB_PACKET_TIMEOUT;
806  } else if(value == 0x00) {
807  gbSgbPacketTimeout = GBSGB_PACKET_TIMEOUT;
808  gbSgbPacketState = GBSGB_RESET;
809  } else {
810  gbSgbPacketState = GBSGB_NONE;
811  gbSgbPacketTimeout = 0;
812  }
813  break;
814  case GBSGB_PACKET_TRANSMIT:
815  if(value == 0) {
816  gbSgbPacketState = GBSGB_RESET;
817  gbSgbPacketTimeout = 0;
818  } else if (value == 0x30){
819  if(gbSgbPacketNBits == 128) {
820  gbSgbPacketNBits = 0;
821  gbSgbPacketByte = 0;
822  gbSgbPacketNumber++;
823  gbSgbPacketTimeout = 0;
824  if(gbSgbPacketNumber == (gbSgbPacket[0] & 7)) {
825  gbSgbCommand();
826  gbSgbPacketNumber = 0;
827  gbSgbPacketState = GBSGB_NONE;
828  gbSgbPacketTimeout = 0;
829  }
830  } else {
831  if(gbSgbPacketNBits < 128) {
832  gbSgbPacket[gbSgbPacketNumber * 16 + gbSgbPacketByte] >>= 1;
833  gbSgbPacket[gbSgbPacketNumber * 16 + gbSgbPacketByte] |= gbSgbBit;
834  gbSgbPacketNBits++;
835  if(!(gbSgbPacketNBits & 7)) {
836  gbSgbPacketByte++;
837  }
838  gbSgbPacketTimeout = GBSGB_PACKET_TIMEOUT;
839  }
840  }
841  } else {
842  if(value == 0x20)
843  gbSgbBit = 0x00;
844  else
845  gbSgbBit = 0x80;
846  gbSgbPacketTimeout = GBSGB_PACKET_TIMEOUT;
847  }
848  gbSgbReadingController = 0;
849  break;
850  default:
851  gbSgbPacketState = GBSGB_NONE;
852  gbSgbPacketTimeout = 0;
853  break;
854  }
855 }
