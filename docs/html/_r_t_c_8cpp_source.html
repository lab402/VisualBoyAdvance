82 bool rtcWrite(u32 address, u16 value)
83 {
84  if(!rtcEnabled)
85  return false;
86 
87  if(address == 0x80000c8) {
88  rtcClockData.byte2 = (u8)value; // enable ?
89  } else if(address == 0x80000c6) {
90  rtcClockData.byte1 = (u8)value; // read/write
91  } else if(address == 0x80000c4) {
92  if(rtcClockData.byte2 & 1) {
93  if(rtcClockData.state == IDLE && rtcClockData.byte0 == 1 && value == 5) {
94  rtcClockData.state = COMMAND;
95  rtcClockData.bits = 0;
96  rtcClockData.command = 0;
97  } else if(!(rtcClockData.byte0 & 1) && (value & 1)) { // bit transfer
98  rtcClockData.byte0 = (u8)value;
99  switch(rtcClockData.state) {
100  case COMMAND:
101  rtcClockData.command |= ((value & 2) >> 1) << (7-rtcClockData.bits);
102  rtcClockData.bits++;
103  if(rtcClockData.bits == 8) {
104  rtcClockData.bits = 0;
105  switch(rtcClockData.command) {
106  case 0x60:
107  // not sure what this command does but it doesn't take parameters
108  // maybe it is a reset or stop
109  rtcClockData.state = IDLE;
110  rtcClockData.bits = 0;
111  break;
112  case 0x62:
113  // this sets the control state but not sure what those values are
114  rtcClockData.state = READDATA;
115  rtcClockData.dataLen = 1;
116  break;
117  case 0x63:
118  rtcClockData.dataLen = 1;
119  rtcClockData.data[0] = 0x40;
120  rtcClockData.state = DATA;
121  break;
122  case 0x64:
123  break;
124  case 0x65:
125  {
126  struct tm *newtime;
127  time_t long_time;
128 
129  time( &long_time ); /* Get time as long integer. */
130  newtime = localtime( &long_time ); /* Convert to local time. */
131 
132  rtcClockData.dataLen = 7;
133  rtcClockData.data[0] = toBCD(newtime->tm_year);
134  rtcClockData.data[1] = toBCD(newtime->tm_mon+1);
135  rtcClockData.data[2] = toBCD(newtime->tm_mday);
136  rtcClockData.data[3] = toBCD(newtime->tm_wday);
137  rtcClockData.data[4] = toBCD(newtime->tm_hour);
138  rtcClockData.data[5] = toBCD(newtime->tm_min);
139  rtcClockData.data[6] = toBCD(newtime->tm_sec);
140  rtcClockData.state = DATA;
141  }
142  break;
143  case 0x67:
144  {
145  struct tm *newtime;
146  time_t long_time;
147 
148  time( &long_time ); /* Get time as long integer. */
149  newtime = localtime( &long_time ); /* Convert to local time. */
150 
151  rtcClockData.dataLen = 3;
152  rtcClockData.data[0] = toBCD(newtime->tm_hour);
153  rtcClockData.data[1] = toBCD(newtime->tm_min);
154  rtcClockData.data[2] = toBCD(newtime->tm_sec);
155  rtcClockData.state = DATA;
156  }
157  break;
158  default:
159  systemMessage(0, N_("Unknown RTC command %02x"), rtcClockData.command);
160  rtcClockData.state = IDLE;
161  break;
162  }
163  }
164  break;
165  case DATA:
166  if(rtcClockData.byte1 & 2) {
167  } else {
168  rtcClockData.byte0 = (rtcClockData.byte0 & ~2) |
169  ((rtcClockData.data[rtcClockData.bits >> 3] >>
170  (rtcClockData.bits & 7)) & 1)*2;
171  rtcClockData.bits++;
172  if(rtcClockData.bits == 8*rtcClockData.dataLen) {
173  rtcClockData.bits = 0;
174  rtcClockData.state = IDLE;
175  }
176  }
177  break;
178  case READDATA:
179  if(!(rtcClockData.byte1 & 2)) {
180  } else {
181  rtcClockData.data[rtcClockData.bits >> 3] =
182  (rtcClockData.data[rtcClockData.bits >> 3] >> 1) |
183  ((value << 6) & 128);
184  rtcClockData.bits++;
185  if(rtcClockData.bits == 8*rtcClockData.dataLen) {
186  rtcClockData.bits = 0;
187  rtcClockData.state = IDLE;
188  }
189  }
190  break;
191  default:
192  break;
193  }
194  } else
195  rtcClockData.byte0 = (u8)value;
196  }
197  }
198  return true;
199 }
