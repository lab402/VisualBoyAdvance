167  if(theApp.cartridgeType == IMAGE_GBA) {
168  frameSkip = nID - ID_OPTIONS_VIDEO_FRAMESKIP_0;
169  } else {

178  case ID_OPTIONS_VIDEO_FRAMESKIP_7:
179  case ID_OPTIONS_VIDEO_FRAMESKIP_8:
180  case ID_OPTIONS_VIDEO_FRAMESKIP_9:

182  frameSkip = 6 + nID - ID_OPTIONS_VIDEO_FRAMESKIP_6;
183  } else {
184  gbFrameSkip = 6 + nID - ID_OPTIONS_VIDEO_FRAMESKIP_6;
185  }
186  if(emulating)

188  theApp.autoFrameSkip = false;

323 void MainWnd::OnOptionsVideoFullscreen()
324 {
325  theApp.winCheckFullscreen();
326  GUID *pGUID = NULL;
327  int size = theApp.display->selectFullScreenMode(&pGUID);
328  if(size != -1) {
329  int width = (size >> 12) & 4095;
330  int height = (size & 4095);
331  int colorDepth = (size >> 24);
332  if(width != theApp.fsWidth ||
333  height != theApp.fsHeight ||
334  colorDepth != theApp.fsColorDepth ||
335  pGUID != theApp.pVideoDriverGUID ||
336  theApp.videoOption != VIDEO_OTHER) {
337  theApp.fsForceChange = true;
338  theApp.fsWidth = width;
339  theApp.fsHeight = height;
340  theApp.fsColorDepth = colorDepth;
341  theApp.pVideoDriverGUID = pGUID;
342  if(pGUID) {
343  theApp.videoDriverGUID = *pGUID;
344  regSetDwordValue("defaultVideoDriver", FALSE);
345  regSetBinaryValue("videoDriverGUID",
346  (char *)pGUID, sizeof(GUID));
347  } else {
348  regSetDwordValue("defaultVideoDriver", TRUE);
349  }
350  theApp.updateVideoSize(ID_OPTIONS_VIDEO_FULLSCREEN);
351  theApp.m_pMainWnd->PostMessage(VBA_CONFIRM_MODE);
352  }
353  }
354  theApp.winAccelMgr.UpdateMenu(theApp.menu);
355 }
