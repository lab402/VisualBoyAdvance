98 char *sdlGetFilename(char *name)
99 {
100  static char filebuffer[2048];
101 
102  int len = strlen(name);
103 
104  char *p = name + len - 1;
105 
106  while(true) {
107  if(*p == '/' ||
108  *p == '\\') {
109  p++;
110  break;
111  }
112  len--;
113  p--;
114  if(len == 0)
115  break;
116  }
117 
118  if(len == 0)
119  strcpy(filebuffer, name);
120  else
121  strcpy(filebuffer, p);
122  return filebuffer;
123 }

155  systemGbPalette[i++] = (0x15) | (0x15 << 5) | (0x15 << 10);
156  systemGbPalette[i++] = (0x0c) | (0x0c << 5) | (0x0c << 10);
157  systemGbPalette[i++] = 0;

162  bool failed = false;
163  if(CPUIsZipFile(szFile)) {
164  unzFile unz = unzOpen(szFile);
165

168  exit(-1);
169  }

171 
172  if(r != UNZ_OK) {
173  unzClose(unz);
174  systemMessage(0, "Bad ZIP file %s", szFile);
175  exit(-1);
176  }
177 
178  bool found = false;
179 
180  unz_file_info info;
181 
182  while(true) {
183  r = unzGetCurrentFileInfo(unz,
184  &info,
185  buffer,
186  sizeof(buffer),
187  NULL,
188  0,
189  NULL,
190  0);
191 
192  if(r != UNZ_OK) {
193  unzClose(unz);
194  systemMessage(0,"Bad ZIP file %s", szFile);
195  exit(-1);
196  }
197 
198  if(utilIsGBImage(buffer)) {
199  found = true;
200  cartridgeType = 1;
201  break;
202  }
203  if(utilIsGBAImage(buffer)) {
204  found = true;
205  cartridgeType = 0;
206  break;
207  }
208 
209  r = unzGoToNextFile(unz);
210 
211  if(r != UNZ_OK)
212  break;
213  }
214 
215  if(!found) {
216  unzClose(unz);
217  systemMessage(0, "No image found on ZIP file %s", szFile);
218  exit(-1);
219  }
220 
221  unzClose(unz);
222  }
223 
224  if(utilIsGBImage(szFile) || cartridgeType == 1) {
225  failed = !gbLoadRom(szFile);
226  cartridgeType = 1;
227  emulator = GBSystem;
228  } else if(utilIsGBAImage(szFile) || cartridgeType == 0) {
229  failed = !CPULoadRom(szFile);
230  cartridgeType = 0;
231  emulator = GBASystem;
232 
233  CPUInit(biosFileName, useBios);
234  CPUReset();
235  } else {
236  systemMessage(0, "Unknown file type %s", szFile);
237  exit(-1);
238  }
239 
240  if(failed) {
241  systemMessage(0, "Failed to load file %s", szFile);
242  exit(-1);
243  }
244  strcpy(filename, szFile);
245  char *p = strrchr(filename, '.');
246 
247  if(p)
248  *p = 0;
249  } else {
250  cartridgeType = 0;
251  strcpy(filename, "gnu_stub");
252  rom = (u8 *)malloc(0x2000000);
253  workRAM = (u8 *)calloc(1, 0x40000);
254  bios = (u8 *)calloc(1,0x4000);
255  internalRAM = (u8 *)calloc(1,0x8000);
256  paletteRAM = (u8 *)calloc(1,0x400);
257  vram = (u8 *)calloc(1, 0x20000);
258  oam = (u8 *)calloc(1, 0x400);
259  pix = (u8 *)calloc(1, 4 * 241 * 162);
260  ioMem = (u8 *)calloc(1, 0x400);
261 
262  emulator = GBASystem;
263 
264  CPUInit(biosFileName, useBios);
265  CPUReset();
266  }
267 
268  if(debuggerStub)
269  remoteInit();
270 
271  if(cartridgeType == 0) {
272  } else if (cartridgeType == 1) {
273  if(gbBorderOn) {
274  gbBorderLineSkip = 256;
275  gbBorderColumnSkip = 48;
276  gbBorderRowSkip = 40;
277  } else {
278  gbBorderLineSkip = 160;
279  gbBorderColumnSkip = 0;
280  gbBorderRowSkip = 0;
281  }
282  } else {
283  }
284 
285  for(int i = 0; i < 0x10000; i++) {
286  systemColorMap32[i] = ((i & 0x1f) << systemRedShift) |
287  (((i & 0x3e0) >> 5) << systemGreenShift) |
288  (((i & 0x7c00) >> 10) << systemBlueShift);
289  }
290 
291  emulating = 1;
292  soundInit();
293 
294  while(emulating) {
295  if(!paused) {
296  if(debugger && emulator.emuHasDebugger)
297  dbgMain();
298  else
299  emulator.emuMain(emulator.emuCount);
300  }
301  }
302  emulating = 0;
303  fprintf(stderr,"Shutting down\n");
304  remoteCleanUp();
305  soundShutdown();
306 
307  if(gbRom != NULL || rom != NULL) {
308  emulator.emuCleanUp();
309  }
310 
311  return 0;
312 }
