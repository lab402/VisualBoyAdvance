213 static BOOL CALLBACK EnumAxesCallback( const DIDEVICEOBJECTINSTANCE* pdidoi,
214  VOID* pContext )
215 {
216  DIPROPRANGE diprg;
217  diprg.diph.dwSize = sizeof(DIPROPRANGE);
218  diprg.diph.dwHeaderSize = sizeof(DIPROPHEADER);
219  diprg.diph.dwHow = DIPH_BYOFFSET;
220  diprg.diph.dwObj = pdidoi->dwOfs; // Specify the enumerated axis
221 
222  diprg.lMin = -32768;
223  diprg.lMax = 32767;
224  // try to set the range
225  if (FAILED(currentDevice->device->SetProperty(DIPROP_RANGE, &diprg.diph))) {
226  // Get the range for the axis
227  if ( FAILED(currentDevice->device->
228  GetProperty( DIPROP_RANGE, &diprg.diph ) ) ) {
229  return DIENUM_STOP;
230  }
231  }
232 
233  DIPROPDWORD didz;
234 
235  didz.diph.dwSize = sizeof(didz);
236  didz.diph.dwHeaderSize = sizeof(DIPROPHEADER);
237  didz.diph.dwHow = DIPH_BYOFFSET;
238  didz.diph.dwObj = pdidoi->dwOfs;
239 
240  didz.dwData = 5000;
241 
242  currentDevice->device->SetProperty(DIPROP_DEADZONE, &didz.diph);
243 
244  LONG center = (diprg.lMin + diprg.lMax)/2;
245  LONG threshold = (diprg.lMax - center)/2;
246 
247  // only 8 axis supported
248  if (axisNumber < 8) {
249  currentDevice->axis[axisNumber].center = center;
250  currentDevice->axis[axisNumber].negative = center - threshold;
251  currentDevice->axis[axisNumber].positive = center + threshold;
252  currentDevice->axis[axisNumber].offset = pdidoi->dwOfs;
253  }
254  axisNumber++;
255  return DIENUM_CONTINUE;
256 }

264 static BOOL CALLBACK DIEnumDevicesCallback(LPCDIDEVICEINSTANCE pInst,
265  LPVOID lpvContext)
266 {
267  ZeroMemory(&pDevices[numDevices],sizeof(deviceInfo));
268 
269  HRESULT hRet = pDirectInput->CreateDevice(pInst->guidInstance,
270  &pDevices[numDevices].device,
271  NULL);
272 
273  if (hRet != DI_OK)
274  return DIENUM_STOP;
275 
276  DIDEVCAPS caps;
277  caps.dwSize=sizeof(DIDEVCAPS);
278 
279  hRet = pDevices[numDevices].device->GetCapabilities(&caps);
280 
281  if (hRet == DI_OK) {
282  if (caps.dwFlags & DIDC_POLLEDDATAFORMAT ||
283  caps.dwFlags & DIDC_POLLEDDEVICE)
284  pDevices[numDevices].isPolled = TRUE;
285 
286  pDevices[numDevices].nButtons = caps.dwButtons;
287  pDevices[numDevices].nAxes = caps.dwAxes;
288  pDevices[numDevices].nPovs = caps.dwPOVs;
289 
290  for (int i = 0; i < 6; i++) {
291  pDevices[numDevices].axis[i].center = 0x8000;
292  pDevices[numDevices].axis[i].negative = 0x4000;
293  pDevices[numDevices].axis[i].positive = 0xc000;
294  }
295  }
296 
297 
298  numDevices++;
299 
300 
301  return DIENUM_CONTINUE;

312 static int getPovState(DWORD value)
313 {
314  int state = 0;
315  if (LOWORD(value) != 0xFFFF) {
316  if (value < 9000 || value > 27000)
317  state |= POV_UP;
318  if (value > 0 && value < 18000)
319  state |= POV_RIGHT;
320  if (value > 9000 && value < 27000)
321  state |= POV_DOWN;
322  if (value > 18000)
323  state |= POV_LEFT;
324  }
325  return state;
326 }

492 static void checkKeyboard()
493 {
494  // mham fix. Patch #1378104
495  UCHAR keystate[256];
496  HRESULT hret = pDevices[0].device->Acquire();
497 
498  if (pDevices[0].first) {
499  pDevices[0].device->GetDeviceState(256, (LPVOID)pDevices[0].data);
500  pDevices[0].first = FALSE;
501  return;
502  }
503 
504  hret = pDevices[0].device->
505  GetDeviceState(256, (LPVOID)keystate);
506 
507  if (hret == DIERR_INPUTLOST || hret == DIERR_NOTACQUIRED) {
508  return;
509  }
510 
511  if (hret == DI_OK) {
512  for (int i = 0; i < 256; i++) {
513  if (keystate[i] == pDevices[0].data[i]) continue;
514  if (KEYDOWN(keystate, i)) {
515  SendMessage(GetFocus(), JOYCONFIG_MESSAGE,0,i);
516  break;
517  }
518  }
519  }
520  memcpy(pDevices[0].data, keystate, sizeof(UCHAR) * 256);
521 }

523 static void checkJoypads()
524 {
525  DIDEVICEOBJECTINSTANCE di;
526 
527  ZeroMemory(&di,sizeof(DIDEVICEOBJECTINSTANCE));
528 
529  di.dwSize = sizeof(DIDEVICEOBJECTINSTANCE);
530 
531  int i =0;
532 
533  DIJOYSTATE joystick;
534 
535  for (i = 1; i < numDevices; i++) {
536  HRESULT hret = pDevices[i].device->Acquire();
537 
538 
539  if (pDevices[i].isPolled)
540  ((LPDIRECTINPUTDEVICE2)pDevices[i].device)->Poll();
541 
542  hret = pDevices[i].device->GetDeviceState(sizeof(joystick), &joystick);
543 
544  int j;
545 
546  if (pDevices[i].first) {
547  memcpy(&pDevices[i].state, &joystick, sizeof(joystick));
548  pDevices[i].first = FALSE;
549  continue;
550  }
551 
552  for (j = 0; j < pDevices[i].nButtons; j++) {
553  if (((pDevices[i].state.rgbButtons[j] ^ joystick.rgbButtons[j])
554  & joystick.rgbButtons[j]) & 0x80) {
555  HWND focus = GetFocus();
556 
557  SendMessage(focus, JOYCONFIG_MESSAGE, i,j+128);
558  }
559  }
560 
561  for (j = 0; j < pDevices[i].nAxes && j < 8; j++) {
562  LONG value = pDevices[i].axis[j].center;
563  LONG old = 0;
564  switch (pDevices[i].axis[j].offset) {
565  case DIJOFS_X:
566  value = joystick.lX;
567  old = pDevices[i].state.lX;
568  break;
569  case DIJOFS_Y:
570  value = joystick.lY;
571  old = pDevices[i].state.lY;
572  break;
573  case DIJOFS_Z:
574  value = joystick.lZ;
575  old = pDevices[i].state.lZ;
576  break;
577  case DIJOFS_RX:
578  value = joystick.lRx;
579  old = pDevices[i].state.lRx;
580  break;
581  case DIJOFS_RY:
582  value = joystick.lRy;
583  old = pDevices[i].state.lRy;
584  break;
585  case DIJOFS_RZ:
586  value = joystick.lRz;
587  old = pDevices[i].state.lRz;
588  break;
589  case DIJOFS_SLIDER(0):
590  value = joystick.rglSlider[0];
591  old = pDevices[i].state.rglSlider[0];
592  break;
593  case DIJOFS_SLIDER(1):
594  value = joystick.rglSlider[1];
595  old = pDevices[i].state.rglSlider[1];
596  break;
597  }
598  if (value != old) {
599  if (value < pDevices[i].axis[j].negative)
600  SendMessage(GetFocus(), JOYCONFIG_MESSAGE, i, (j<<1));
601  else if (value > pDevices[i].axis[j].positive)
602  SendMessage(GetFocus(), JOYCONFIG_MESSAGE, i, (j<<1)+1);
603  }
604  }
605 
606  for (j = 0;j < 4 && j < pDevices[i].nPovs; j++) {
607  if (LOWORD(pDevices[i].state.rgdwPOV[j]) != LOWORD(joystick.rgdwPOV[j])) {
608  int state = getPovState(joystick.rgdwPOV[j]);
609 
610  if (state & POV_UP)
611  SendMessage(GetFocus(), JOYCONFIG_MESSAGE, i, (j<<2)+0x20);
612  else if (state & POV_DOWN)
613  SendMessage(GetFocus(), JOYCONFIG_MESSAGE, i, (j<<2)+0x21);
614  else if (state & POV_RIGHT)
615  SendMessage(GetFocus(), JOYCONFIG_MESSAGE, i, (j<<2)+0x22);
616  else if (state & POV_LEFT)
617  SendMessage(GetFocus(), JOYCONFIG_MESSAGE, i, (j<<2)+0x23);
618  }
619  }
620 
621  memcpy(&pDevices[i].state, &joystick, sizeof(joystick));
622  }
623 }

625 BOOL checkKey(LONG_PTR key)
626 {
627  LONG_PTR dev = (key >> 8);
628 
629  LONG_PTR k = (key & 255);
630 
631  if (dev == 0) {
632  return KEYDOWN(pDevices[0].data,k);
633  } else {
634  if (k < 16) {
635  LONG_PTR axis = k >> 1;
636  LONG value = pDevices[dev].axis[axis].center;
637  switch (pDevices[dev].axis[axis].offset) {
638  case DIJOFS_X:
639  value = pDevices[dev].state.lX;
640  break;
641  case DIJOFS_Y:
642  value = pDevices[dev].state.lY;
643  break;
644  case DIJOFS_Z:
645  value = pDevices[dev].state.lZ;
646  break;
647  case DIJOFS_RX:
648  value = pDevices[dev].state.lRx;
649  break;
650  case DIJOFS_RY:
651  value = pDevices[dev].state.lRy;
652  break;
653  case DIJOFS_RZ:
654  value = pDevices[dev].state.lRz;
655  break;
656  case DIJOFS_SLIDER(0):
657  value = pDevices[dev].state.rglSlider[0];
658  break;
659  case DIJOFS_SLIDER(1):
660  value = pDevices[dev].state.rglSlider[1];
661  break;
662  }
663 
664  if (k & 1)
665  return value > pDevices[dev].axis[axis].positive;

809 u32 DirectInput::readDevice(int which)
810 {
811  u32 res = 0;
812  int i = theApp.joypadDefault;
813  if (which >= 0 && which <= 3)
814  i = which;
815 
816  if (checkKey(joypad[i][KEY_BUTTON_A]))
817  res |= 1;
818  if (checkKey(joypad[i][KEY_BUTTON_B]))
819  res |= 2;
820  if (checkKey(joypad[i][KEY_BUTTON_SELECT]))
821  res |= 4;
822  if (checkKey(joypad[i][KEY_BUTTON_START]))
823  res |= 8;
824  if (checkKey(joypad[i][KEY_RIGHT]))
825  res |= 16;
826  if (checkKey(joypad[i][KEY_LEFT]))
827  res |= 32;
828  if (checkKey(joypad[i][KEY_UP]))
829  res |= 64;
830  if (checkKey(joypad[i][KEY_DOWN]))
831  res |= 128;
832  if (checkKey(joypad[i][KEY_BUTTON_R]))
833  res |= 256;
834  if (checkKey(joypad[i][KEY_BUTTON_L]))
835  res |= 512;
836 
837  if (checkKey(joypad[i][KEY_BUTTON_GS]))
838  res |= 4096;
839 
840  if (theApp.autoFire) {
841  res &= (~theApp.autoFire);
842  if (theApp.autoFireToggle)
843  res |= theApp.autoFire;
844  theApp.autoFireToggle = !theApp.autoFireToggle;
845  }
846 
847  // disallow L+R or U+D of being pressed at the same time
848  if ((res & 48) == 48)
849  res &= ~16;
850  if ((res & 192) == 192)
851  res &= ~128;
852 
853  if (theApp.movieRecording) {
854  if (i == theApp.joypadDefault) {
855  if (res != theApp.movieLastJoypad) {
856  fwrite(&theApp.movieFrame, 1, sizeof(theApp.movieFrame), theApp.movieFile);
857  fwrite(&res, 1, sizeof(res), theApp.movieFile);
858  theApp.movieLastJoypad = res;
859  }
860  }
861  }
862  if (theApp.moviePlaying) {
863  if (theApp.movieFrame == theApp.moviePlayFrame) {
864  theApp.movieLastJoypad = theApp.movieNextJoypad;
865  theApp.movieReadNext();
866  }
867  res = theApp.movieLastJoypad;
868  }
869  // we don't record speed up or screen capture buttons
870  if (checkKey(joypad[i][KEY_BUTTON_SPEED]) || theApp.speedupToggle)
871  res |= 1024;
872  if (checkKey(joypad[i][KEY_BUTTON_CAPTURE]))
873  res |= 2048;
874 
875  return res;
876 }
