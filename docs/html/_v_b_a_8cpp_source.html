387 /* Parse a command line buffer into arguments */
388 static int parseCommandLine(char *cmdline, char **argv)
389 {
390  char *bufp;
391  int argc;
392 
393  argc = 0;
394  for ( bufp = cmdline; *bufp; ) {
395  /* Skip leading whitespace */
396  while ( isspace(*bufp) ) {
397  ++bufp;
398  }
399  /* Skip over argument */
400  if ( *bufp == '"' ) {
401  ++bufp;
402  if ( *bufp ) {
403  if ( argv ) {
404  argv[argc] = bufp;
405  }
406  ++argc;
407  }
408  /* Skip over word */
409  while ( *bufp && (*bufp != '"') ) {
410  ++bufp;
411  }
412  } else {
413  if ( *bufp ) {
414  if ( argv ) {
415  argv[argc] = bufp;
416  }
417  ++argc;
418  }
419  /* Skip over word */
420  while ( *bufp && ! isspace(*bufp) ) {
421  ++bufp;
422  }
423  }
424  if ( *bufp ) {
425  if ( argv ) {
426  *bufp = '\0';
427  }
428  ++bufp;
429  }
430  }
431  if ( argv ) {
432  argv[argc] = NULL;
433  }
434  return(argc);
435 }

558  }
559

561  dest.bottom = point.y;
562  dest.right = point.x;
563 
564  // make sure that dest rect lies in the monitor
565  if(videoOption >= VIDEO_320x240) {
566  dest.top -= windowPositionY;
567  dest.left -= windowPositionX;
568  dest.bottom-= windowPositionY;
569  dest.right -= windowPositionX;
570  }
571 
572  if(skin)
573  return;
574 
575  int menuSkip = 0;
576 
577  if(videoOption >= VIDEO_320x240 && menuToggle) {
578  int m = GetSystemMetrics(SM_CYMENU);
