121 void MapView::renderTextScreen(u16 control)
122 {
123  u16 *palette = (u16 *)paletteRAM;
124  u8 *charBase = &vram[((control >> 2) & 0x03) * 0x4000];
125  u16 *screenBase = (u16 *)&vram[((control >> 8) & 0x1f) * 0x800];
126  u8 *bmp = data;
127 
128  int sizeX = 256;
129  int sizeY = 256;
130  switch((control >> 14) & 3) {
131  case 0:
132  break;
133  case 1:
134  sizeX = 512;
135  break;
136  case 2:
137  sizeY = 512;
138  break;
139  case 3:
140  sizeX = 512;
141  sizeY = 512;
142  break;
143  }
144 
145  w = sizeX;
146  h = sizeY;
147 
148  if(control & 0x80) {
149  for(int y = 0; y < sizeY; y++) {
150  int yy = y & 255;
151 
152  if(y == 256 && sizeY > 256) {
153  screenBase += 0x400;
154  if(sizeX > 256)
155  screenBase += 0x400;
156  }
157  u16 *screenSource = screenBase + ((yy>>3)*32);
158 
159  for(int x = 0; x < sizeX; x++) {
160  u16 data = *screenSource;
161 
162  int tile = data & 0x3FF;
163  int tileX = (x & 7);
164  int tileY = y & 7;
165 
166  if(data & 0x0400)
167  tileX = 7 - tileX;
168  if(data & 0x0800)
169  tileY = 7 - tileY;
170 
171  u8 c = charBase[tile * 64 + tileY * 8 + tileX];
172 
173  u16 color = palette[c];
174 
175  *bmp++ = ((color >> 10) & 0x1f) << 3;
176  *bmp++ = ((color >> 5) & 0x1f) << 3;
177  *bmp++ = (color & 0x1f) << 3;
178 
179  if(data & 0x0400) {
180  if(tileX == 0)
181  screenSource++;
182  } else if(tileX == 7)
183  screenSource++;
184  if(x == 255 && sizeX > 256) {
185  screenSource = screenBase + 0x400 + ((yy>>3)*32);
186  }
187  }
188  }
189  } else {
190  for(int y = 0; y < sizeY; y++) {
191  int yy = y & 255;
192 
193  if(y == 256 && sizeY > 256) {
194  screenBase += 0x400;
195  if(sizeX > 256)
196  screenBase += 0x400;
197  }

209  if(data & 0x0800)

211 
212  u8 color = charBase[tile * 32 + tileY * 4 + (tileX>>1)];
213 
214  if(tileX & 1) {
215  color = (color >> 4);
216  } else {
217  color &= 0x0F;
218  }
219 
220  int pal = (*screenSource>>8) & 0xF0;
221  u16 color2 = palette[pal + color];
222 
223  *bmp++ = ((color2 >> 10) & 0x1f) << 3;
224  *bmp++ = ((color2 >> 5) & 0x1f) << 3;
225  *bmp++ = (color2 & 0x1f) << 3;
226

280  u8 *charBase = &vram[((control >> 2) & 0x03) * 0x4000];

282  u8 *bmp = data;
283 
284  int sizeX = 128;
285  int sizeY = 128;
286  switch((control >> 14) & 3) {
287  case 0:
288  break;
289  case 1:
290  sizeX = sizeY = 256;
291  break;
292  case 2:
293  sizeX = sizeY = 512;
294  break;
295  case 3:
296  sizeX = sizeY = 1024;
297  break;
298  }
299 
300  w = sizeX;
301  h = sizeY;
302 
303  if(control & 0x80) {
304  for(int y = 0; y < sizeY; y++) {
305  for(int x = 0; x < sizeX; x++) {
306  int tile = screenBase[(x>>3) + (y>>3)*(w>>3)];
307 
308  int tileX = (x & 7);
309  int tileY = y & 7;
310 
311  u8 color = charBase[tile * 64 + tileY * 8 + tileX];
312  u16 color2 = palette[color];
313 
314  *bmp++ = ((color2 >> 10) & 0x1f) << 3;
315  *bmp++ = ((color2 >> 5) & 0x1f) << 3;
316  *bmp++ = (color2 & 0x1f) << 3;
317  }
318  }
319  } else {
320  for(int y = 0; y < sizeY; y++) {
321  for(int x = 0; x < sizeX; x++) {
322  int tile = screenBase[(x>>3) + (y>>3)*(w>>3)];
323 
324  int tileX = (x & 7);
325  int tileY = y & 7;
326 
327  u8 color = charBase[tile * 64 + tileY * 8 + tileX];
328  u16 color2 = palette[color];
329 
330  *bmp++ = ((color2 >> 10) & 0x1f) << 3;
331  *bmp++ = ((color2 >> 5) & 0x1f) << 3;
332  *bmp++ = (color2 & 0x1f) << 3;
333  }
334  }
335  }
336 
337  u32 xx;
338  u32 yy;
339 
340  switch(bg) {
341  case 2:
342  xx = BG2X_L | BG2X_H << 16;
343  yy = BG2Y_L | BG2Y_H << 16;
344 
345  /*
346  renderView(xx, yy,
347  BG2PA, BG2PC,
348  BG2PB, BG2PD,
349  (sizeX -1) <<8,
350  (sizeY -1) << 8,
351  (control & 0x2000) != 0);
352  */
353  break;
354  case 3:
355  xx = BG3X_L | BG3X_H << 16;
356  yy = BG3Y_L | BG3Y_H << 16;
357  /*
358  renderView(xx, yy,
359  BG3PA, BG3PC,
360  BG3PB, BG3PD,
361  (sizeX -1) <<8,
362  (sizeY -1) << 8,

365  break;
366  }

474 void MapView::paint()
475 {
476  if(vram == NULL)
477  return;
478  int mode = DISPCNT & 7;
479 
480  switch(bg) {
481  default:
482  case 0:
483  control = BG0CNT;
484  break;
485  case 1:
486  control = BG1CNT;
487  break;
488  case 2:
489  control = BG2CNT;
490  break;
491  case 3:
492  control = BG3CNT;
493  break;
494  }
495 
496  switch(mode) {
497  case 0:
498  renderMode0();
499  break;
500  case 1:
501  renderMode1();
502  break;
503  case 2:
504  renderMode2();
505  break;
506  case 3:
507  renderMode3();
508  break;
509  case 4:
510  renderMode4();
511  break;
512  case 5:
513  renderMode5();
514  break;
515  case 6:
516  renderMode5();
517  break;
518  case 7:
519  renderMode5();
520  break;
521  }
522  enableButtons(mode);
523  SIZE s;
524 
525  if(mapView.getStretch()) {
526  mapView.setSize(w, h);
527  s.cx = s.cy = 1;
528  mapView.SetScrollSizes(MM_TEXT, s);
529  } else {
530  mapView.setSize(w, h);
531  s.cx = w;
532  s.cy = h;
533  mapView.SetScrollSizes(MM_TEXT, s);
534  }
535 
536  mapView.refresh();
537 
538  CString buffer;
539 
540  u32 charBase = ((control >> 2) & 0x03) * 0x4000 + 0x6000000;
541  u32 screenBase = ((control >> 8) & 0x1f) * 0x800 + 0x6000000;
542 
543  buffer.Format("%d", mode);
544  m_mode.SetWindowText(buffer);
545 
546  if(mode >= 3) {
547  m_mapbase.SetWindowText("");
548  m_charbase.SetWindowText("");
549  } else {
550  buffer.Format("0x%08X", screenBase);
551  m_mapbase.SetWindowText(buffer);
552 
553  buffer.Format("0x%08X", charBase);
554  m_charbase.SetWindowText(buffer);
555  }
556 
557  buffer.Format("%dx%d", w, h);
558  m_dim.SetWindowText(buffer);
559 
560  m_numcolors.SetWindowText(control & 0x80 ? "256" : "16");
561 
562  buffer.Format("%d", control & 3);
563  m_priority.SetWindowText(buffer);
564 
565  m_mosaic.SetWindowText(control & 0x40 ? "1" : "0");
566 
567  m_overflow.SetWindowText(bg <= 1 ? "" :
568  control & 0x2000 ? "1" : "0");
569 }

744 u32 MapView::GetTextClickAddress(u32 base, int x, int y)
745 {
746  if(y > 255 && h > 256) {
747  base += 0x800;
748  if(w > 256)
749  base += 0x800;
750  }
751  if(x >= 256)
752  base += 0x800;
753  x &= 255;
754  y &= 255;
755  base += (x>>3)*2 + 64*(y>>3);
756 
757  return base;
758 }
