66 BOOL WINAPI
67 DDEnumCallbackEx(GUID *pGUID, LPSTR pDescription, LPSTR pName, LPVOID pContext, HMONITOR hm)
68 {
69  if (pGUID)
70  {
71  Drivers[gDriverCnt].GUIDcopy = *pGUID;
72  Drivers[gDriverCnt].pGUID = &Drivers[gDriverCnt].GUIDcopy;
73  }
74  else
75  Drivers[gDriverCnt].pGUID = NULL;
76  Drivers[gDriverCnt].szDescription[127] = '\0';
77  Drivers[gDriverCnt].szName[127] = '\0';
78  strncpy(Drivers[gDriverCnt].szDescription,pDescription,127);
79  strncpy(Drivers[gDriverCnt].szName,pName,127);
80  Drivers[gDriverCnt].hm = hm;
81  if (gDriverCnt < MAX_DRIVERS)
82  gDriverCnt++;
83  else
84  return DDENUMRET_CANCEL;
85  return DDENUMRET_OK;
86 }

101 static HRESULT WINAPI addVideoMode(LPDDSURFACEDESC2 surf, LPVOID lpContext)
102 {
103  HWND h = (HWND)lpContext;
104  char buffer[50];
105 
106  switch( surf->ddpfPixelFormat.dwRGBBitCount )
107  {
108  case 16:
109  case 24:
110  case 32:
111  sprintf(
112  buffer,
113  _T("%4dx%4dx%2d"),
114  surf->dwWidth,
115  surf->dwHeight,
116  surf->ddpfPixelFormat.dwRGBBitCount
117  );
118  WPARAM pos = ::SendMessage( h, LB_ADDSTRING, 0, (LPARAM)buffer );
119  ::SendMessage(
120  h,
121  LB_SETITEMDATA,
122  pos,
123  (surf->ddpfPixelFormat.dwRGBBitCount << 24) |
124  ((surf->dwWidth & 4095) << 12) |
125  (surf->dwHeight & 4095)
126  );
127  break;
128  }
129 
130  return DDENUMRET_OK;
131 }

133 int winVideoModeSelect(CWnd *pWnd, GUID **guid)
134 {
135 #ifdef _AFXDLL
136  HINSTANCE h = AfxLoadLibrary("ddraw.dll");
137 #else
138  HMODULE h = LoadLibrary( _T("ddraw.dll") );
139 #endif
140 
141  // If ddraw.dll doesn't exist in the search path,
142  // then DirectX probably isn't installed, so fail.
143  if (!h)
144  return -1;
145 
146  gDriverCnt = 0;
147 
148  // Note that you must know which version of the
149  // function to retrieve (see the following text).
150  // For this example, we use the ANSI version.
151  LPDIRECTDRAWENUMERATEEX lpDDEnumEx;
152  lpDDEnumEx = (LPDIRECTDRAWENUMERATEEX)
153  GetProcAddress(h,"DirectDrawEnumerateExA");
154 
155  // If the function is there, call it to enumerate all display
156  // devices attached to the desktop, and any non-display DirectDraw
157  // devices.
158  if (lpDDEnumEx)
159  lpDDEnumEx(DDEnumCallbackEx, NULL,
160  DDENUM_ATTACHEDSECONDARYDEVICES |
161  DDENUM_NONDISPLAYDEVICES
162  );
163  else {
164  /*
165  * We must be running on an old version of DirectDraw.
166  * Therefore MultiMon isn't supported. Fall back on
167  * DirectDrawEnumerate to enumerate standard devices on a
168  * single-monitor system.
169  */
170  BOOL (WINAPI *lpDDEnum)(LPDDENUMCALLBACK, LPVOID);
171 
172  lpDDEnum = (BOOL (WINAPI *)(LPDDENUMCALLBACK, LPVOID))
173  GetProcAddress(h, "DirectDrawEnumerateA");
174  if(lpDDEnum)
175  lpDDEnum(DDEnumCallback,NULL);
176 
177  /* Note that it could be handy to let the OldCallback function
178  * be a wrapper for a DDEnumCallbackEx.
179  *
180  * Such a function would look like:
181  * BOOL FAR PASCAL OldCallback(GUID FAR *lpGUID,
182  * LPSTR pDesc,
183  * LPSTR pName,
184  * LPVOID pContext)
185  * {
186  * return Callback(lpGUID,pDesc,pName,pContext,NULL);
187  * }
188  */
189  }
190 
191  int selected = 0;
192 
193  if(gDriverCnt > 1) {
194  VideoDriverSelect d(pWnd);
195 
196  INT_PTR selected = d.DoModal();
197 
198  if(selected == -1) {
199 #ifdef _AFXDLL
200  AfxFreeLibrary( h );
201 #else
202  FreeLibrary( h );
203 #endif
204 
205  return -1;
206  }
207  }
208 
209  HRESULT (WINAPI *DDrawCreateEx)(GUID *,LPVOID *,REFIID,IUnknown *);
210  DDrawCreateEx = (HRESULT (WINAPI *)(GUID *,LPVOID *,REFIID,IUnknown *))
211  GetProcAddress(h, "DirectDrawCreateEx");
212 
213  LPDIRECTDRAW7 ddraw = NULL;
214  if(DDrawCreateEx) {
215  HRESULT hret = DDrawCreateEx(Drivers[selected].pGUID,
216  (void **)&ddraw,
217  IID_IDirectDraw7,
218  NULL);
219  if(hret != DD_OK) {
220  systemMessage(0, "Error during DirectDrawCreateEx: %08x", hret);
221 #ifdef _AFXDLL
222  AfxFreeLibrary( h );
223 #else
224  FreeLibrary( h );
225 #endif
226  return -1;
227  }
228  } else {
229  // should not happen....
230  systemMessage(0, "Error getting DirectDrawCreateEx");
231 #ifdef _AFXDLL
232  AfxFreeLibrary( h );
233 #else
234  FreeLibrary( h );
235 #endif
236  return -1;
237  }
238 
239  VideoMode dlg(ddraw, pWnd);
240 
241  INT_PTR res = dlg.DoModal();
242 
243  if(res != -1) {
244  *guid = Drivers[selected].pGUID;
245  }
246  ddraw->Release();
247  ddraw = NULL;
248 
249  // If the library was loaded by calling LoadLibrary(),
250  // then you must use FreeLibrary() to let go of it.
251 #ifdef _AFXDLL
252  AfxFreeLibrary( h );
253 #else
254  FreeLibrary( h );
255 #endif
256 
257  return (int)res;
258 }
