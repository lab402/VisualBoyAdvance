236 void Disassemble::OnVScroll(UINT nSBCode, UINT nPos, CScrollBar* pScrollBar)
237 {
238  switch(nSBCode) {
239  case SB_LINEDOWN:
240  if(mode == 0) {
241  if(armState)
242  address += 4;
243  else
244  address += 2;
245  } else if(mode == 1)
246  address += 4;
247  else
248  address += 2;
249  break;
250  case SB_LINEUP:
251  if(mode == 0) {
252  if(armState)
253  address -= 4;
254  else
255  address -= 2;
256  } else if(mode == 1)
257  address -= 4;
258  else
259  address -= 2;
260  break;
261  case SB_PAGEDOWN:
262  if(mode == 0) {
263  if(armState)
264  address += count*4;
265  else
266  address += count*2;
267  } else if(mode == 1)
268  address += count*4;
269  else
270  address += count*2;
271  break;
272  case SB_PAGEUP:
273  if(mode == 0) {
274  if(armState)
275  address -= count*4;
276  else
277  address -= count*2;
278  } else if(mode == 1)
279  address -= count*4;
280  else
281  address -= count*2;
282  break;
283  }
284  refresh();
285 
286  CDialog::OnVScroll(nSBCode, nPos, pScrollBar);
287 }

289 void Disassemble::refresh()
290 {
291  if(rom == NULL)
292  return;
293 
294  bool arm = armState;
295 
296  if(mode != 0) {
297  if(mode == 1)
298  arm = true;
299  else
300  arm = false;
301  }
302 
303  int h = m_list.GetItemHeight(0);
304  RECT r;
305  m_list.GetClientRect(&r);
306  count = (r.bottom - r.top+1)/h;
307 
308  m_list.ResetContent();
309  if(!emulating && theApp.cartridgeType == 0)
310  return;
311 
312  char buffer[80];
313  u32 addr = address;
314  int i;
315  int sel = -1;
316  for(i = 0; i < count; i++) {
317  if(addr == armNextPC)
318  sel = i;
319  if(arm) {
320  addr += disArm(addr, buffer, 3);
321  } else {
322  addr += disThumb(addr, buffer, 3);
323  }
324  m_list.InsertString(-1, buffer);
325  }
326 
327  if(sel != -1)
328  m_list.SetCurSel(sel);
329 
330  CPUUpdateCPSR();
331 
332  for(i = 0; i < 17; i++) {
333  sprintf(buffer, "%08x", reg[i].I);
334  GetDlgItem(IDC_R0+i)->SetWindowText(buffer);
335  }
336 
337  m_n = (reg[16].I & 0x80000000) != 0;
338  m_z = (reg[16].I & 0x40000000) != 0;
339  m_c = (reg[16].I & 0x20000000) != 0;
340  m_v = (reg[16].I & 0x10000000) != 0;
341  m_i = (reg[16].I & 0x80) != 0;
342  m_f = (reg[16].I & 0x40) != 0;
343  m_t = (reg[16].I & 0x20) != 0;
344 
345  UpdateData(FALSE);
346 
347  int v = reg[16].I & 0x1f;
348  sprintf(buffer, "%02x", v);
349  GetDlgItem(IDC_MODE)->SetWindowText(buffer);
350 }
