585 int cheatsCheckKeys(u32 keys, u32 extended)
586 {
587  bool onoff = true;
588  int ticks = 0;
589  int i;
590  mastercode = 0;
591 
592  for (i = 0; i<4; i++)
593  if (rompatch2addr [i] != 0) {
594  CHEAT_PATCH_ROM_16BIT(rompatch2addr [i],rompatch2oldval [i]);
595  rompatch2addr [i] = 0;
596  }
597 
598  for (i = 0; i < cheatsNumber; i++) {
599  if(!cheatsList[i].enabled) {
600  // make sure we skip other lines in this code
601  i += getCodeLength(i)-1;
602  continue;
603  }
604  switch(cheatsList[i].size) {
605  case GSA_CODES_ON:
606  onoff = true;
607  break;
608  case GSA_SLOWDOWN:
609  // check if button was pressed and released, if so toggle our state
610  if((cheatsList[i].status & 4) && !(extended & 4))
611  cheatsList[i].status ^= 1;
612  if(extended & 4)
613  cheatsList[i].status |= 4;
614  else
615  cheatsList[i].status &= ~4;
616 
617  if(cheatsList[i].status & 1)
618  ticks += ((cheatsList[i].value & 0xFFFF) * 7);
619  break;
620  case GSA_8_BIT_SLIDE:
621  i++;
622  if(i < cheatsNumber) {
623  u32 addr = cheatsList[i-1].value;
624  u8 value = cheatsList[i].rawaddress;
625  int vinc = (cheatsList[i].value >> 24) & 255;
626  int count = (cheatsList[i].value >> 16) & 255;
627  int ainc = (cheatsList[i].value & 0xffff);
628  while(count > 0) {
629  CPUWriteByte(addr, value);
630  value += vinc;
631  addr += ainc;
632  count--;
633  }
634  }
635  break;
636  case GSA_16_BIT_SLIDE:
637  i++;
638  if(i < cheatsNumber) {
639  u32 addr = cheatsList[i-1].value;
640  u16 value = cheatsList[i].rawaddress;
641  int vinc = (cheatsList[i].value >> 24) & 255;
642  int count = (cheatsList[i].value >> 16) & 255;
643  int ainc = (cheatsList[i].value & 0xffff)*2;
644  while(count > 0) {
645  CPUWriteHalfWord(addr, value);
646  value += vinc;
647  addr += ainc;
648  count--;
649  }
650  }
651  break;
652  case GSA_32_BIT_SLIDE:
653  i++;
654  if(i < cheatsNumber) {
655  u32 addr = cheatsList[i-1].value;
656  u32 value = cheatsList[i].rawaddress;
657  int vinc = (cheatsList[i].value >> 24) & 255;
658  int count = (cheatsList[i].value >> 16) & 255;
659  int ainc = (cheatsList[i].value & 0xffff)*4;
660  while(count > 0) {
661  CPUWriteMemory(addr, value);
662  value += vinc;
663  addr += ainc;
664  count--;
665  }
666  }
667  break;
668  case GSA_8_BIT_GS_WRITE2:
669  i++;
670  if(i < cheatsNumber) {
671  if(extended & 4) {
672  CPUWriteByte(cheatsList[i-1].value, cheatsList[i].address);
673  }
674  }
675  break;
676  case GSA_16_BIT_GS_WRITE2:
677  i++;
678  if(i < cheatsNumber) {
679  if(extended & 4) {
680  CPUWriteHalfWord(cheatsList[i-1].value, cheatsList[i].address);
681  }
682  }
683  break;
684  case GSA_32_BIT_GS_WRITE2:
685  i++;
686  if(i < cheatsNumber) {
687  if(extended & 4) {
688  CPUWriteMemory(cheatsList[i-1].value, cheatsList[i].address);
689  }
690  }
691  break;
692  case GSA_16_BIT_ROM_PATCH:
693  if((cheatsList[i].status & 1) == 0) {
694  if(CPUReadHalfWord(cheatsList[i].address) != cheatsList[i].value) {
695  cheatsList[i].oldValue = CPUReadHalfWord(cheatsList[i].address);
696  cheatsList[i].status |= 1;
697  CHEAT_PATCH_ROM_16BIT(cheatsList[i].address, cheatsList[i].value);
698  }
699  }
700  break;
701  case GSA_16_BIT_ROM_PATCH2C:
702  i++;
703  if(i < cheatsNumber) {
704  rompatch2addr [0] = ((cheatsList[i-1].value & 0x00FFFFFF) << 1) + 0x8000000;
705  rompatch2oldval [0] = CPUReadHalfWord(rompatch2addr [0]);
706  rompatch2val [0] = cheatsList[i].rawaddress & 0xFFFF;
707  }
708  break;
709  case GSA_16_BIT_ROM_PATCH2D:
710  i++;
711  if(i < cheatsNumber) {
712  rompatch2addr [1] = ((cheatsList[i-1].value & 0x00FFFFFF) << 1) + 0x8000000;
713  rompatch2oldval [1] = CPUReadHalfWord(rompatch2addr [1]);
714  rompatch2val [1] = cheatsList[i].rawaddress & 0xFFFF;
715  }
716  break;

719  if(i < cheatsNumber) {

721  rompatch2oldval [2] = CPUReadHalfWord(rompatch2addr [2]);

724  break;

728  rompatch2addr [3] = ((cheatsList[i-1].value & 0x00FFFFFF) << 1) + 0x8000000;
729  rompatch2oldval [3] = CPUReadHalfWord(rompatch2addr [3]);
730  rompatch2val [3] = cheatsList[i].rawaddress & 0xFFFF;
731  }
732  break;
733  case MASTER_CODE:

760  CPUWriteMemory(cheatsList[i].address, cheatsList[i].value);
761  }
762  break;
763  case CBA_IF_KEYS_PRESSED:
764  {
765  u16 value = cheatsList[i].value;
766  u32 addr = cheatsList[i].address;
767  if((addr & 0xF0) == 0x20) {
768  if((keys & value) == 0) {
769  i++;
770  }
771  } else if((addr & 0xF0) == 0x10) {
772  if((keys & value) == value) {
773  i++;
774  }
775  } else if((addr & 0xF0) == 0x00) {
776  if(((~keys) & 0x3FF) == value) {
777  i++;
778  }
779  }
780  }

784  i++;
785  }

790  u16 value = cheatsList[i].value;
791  i++;

795  int inc = cheatsList[i].value;

812  cheatsList[i].value);
