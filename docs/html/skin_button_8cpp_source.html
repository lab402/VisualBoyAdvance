95 void SkinButton::OnPaint()
96 {
97  PAINTSTRUCT ps;
98  HDC hDC = ::BeginPaint(m_hWnd, &ps);
99  HDC memDC = ::CreateCompatibleDC(hDC);
100  LRESULT state = ::SendMessage(m_hWnd, BM_GETSTATE, 0, 0);
101  HBITMAP oldBitmap;
102  if(state & BST_PUSHED)
103  oldBitmap = (HBITMAP)SelectObject(memDC, downBmp);
104  else if(mouseOver && overBmp != NULL)
105  oldBitmap = (HBITMAP)SelectObject(memDC, overBmp);
106  else
107  oldBitmap = (HBITMAP)SelectObject(memDC, normalBmp);
108  SelectClipRgn(hDC, region);
109  BitBlt(hDC, 0, 0, theApp.rect.right - theApp.rect.left,
110  theApp.rect.bottom - theApp.rect.top, memDC, 0, 0, SRCCOPY);
111  SelectClipRgn(hDC, NULL);
112  SelectObject(memDC, oldBitmap);
113  DeleteDC(memDC);
114 
115  ::EndPaint(m_hWnd, &ps);
116 }

118 LRESULT SkinButton::OnLButtonUpMsg(WPARAM wParam, LPARAM lParam)
119 {
120  POINT pt;
121  pt.x = LOWORD(lParam);
122  pt.y = HIWORD(lParam);
123  RECT r;
124  GetClientRect(&r);
125  BOOL inside = PtInRect(&r, pt);
126  if(region != NULL)
127  inside &= PtInRegion(region, pt.x, pt.y);
128  if(inside) {
129  ReleaseCapture();
130  Invalidate();
131  HWND hWnd = m_hWnd;
132  if(idCommand != 0)
133  GetParent()->SendMessage(WM_COMMAND, idCommand, 0);
134  else if(buttonMask)
135  theApp.skinButtons = 0;
136  else if(menu != -1) {
137  HMENU m = GetSubMenu(theApp.menu, menu);
138  pt.x = r.left;
139  pt.y = r.bottom;
140  ClientToScreen(&pt);
141  theApp.m_pMainWnd->SendMessage(WM_INITMENUPOPUP, (WPARAM)m, menu);
142  TrackPopupMenu(m, 0, pt.x, pt.y, 0, *theApp.m_pMainWnd, NULL);
143  }
144 
145  return ::DefWindowProc(hWnd, WM_LBUTTONUP, wParam, lParam);
146  }
147  return GetParent()->SendMessage(WM_LBUTTONUP, wParam, lParam);
148 }

150 LRESULT SkinButton::OnLButtonDownMsg(WPARAM wParam, LPARAM lParam)
151 {

153  return Default();
