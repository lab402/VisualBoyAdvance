88 BOOL Directories::OnInitDialog()
89 {

92  CString p = regQueryStringValue("romdir", NULL);
93  if(!p.IsEmpty()) {
94  int len = p.GetLength();
95  if(len > 0)
96  if(p[len-1] == '\\')
97  p = p.Left(len-1);
98  GetDlgItem(IDC_ROM_PATH)->SetWindowText(p);
99  }
100 
101  p = regQueryStringValue("gbromdir", NULL);
102  if(!p.IsEmpty()) {
103  int len = p.GetLength();
104  if(len > 0)
105  if(p[len-1] == '\\')
106  p = p.Left(len-1);
107  GetDlgItem(IDC_GBROM_PATH)->SetWindowText(p);
108  }
109 
110  p = regQueryStringValue("batteryDir", NULL);
111  if(!p.IsEmpty())
112  GetDlgItem(IDC_BATTERY_PATH)->SetWindowText( p);
113  p = regQueryStringValue("saveDir", NULL);
114  if(!p.IsEmpty())
115  GetDlgItem(IDC_SAVE_PATH)->SetWindowText(p);
116  p = regQueryStringValue("captureDir", NULL);
117  if(!p.IsEmpty())
118  GetDlgItem(IDC_CAPTURE_PATH)->SetWindowText(p);
119 
120  CenterWindow();
121 
122  return TRUE; // return TRUE unless you set the focus to a control

275 CString Directories::browseForDir(CString title)
276 {
277  static char buffer[1024];
278  LPMALLOC pMalloc;
279  LPITEMIDLIST pidl;
280 
281  CString res;
282 
283  if(SUCCEEDED(SHGetMalloc(&pMalloc))) {
284  BROWSEINFO bi;
285  ZeroMemory(&bi, sizeof(bi));
286  bi.hwndOwner = m_hWnd;
287  bi.lpszTitle = title;
288  bi.pidlRoot = 0;
289  bi.ulFlags = BIF_RETURNONLYFSDIRS;
290  bi.lpfn = browseCallbackProc;
291  bi.lParam = (LPARAM)(LPCTSTR)initialFolderDir;
292 
293  pidl = SHBrowseForFolder(&bi);
294 
295  if(pidl) {
296  if(SHGetPathFromIDList(pidl, buffer)) {
297  res = buffer;
298  }
299  pMalloc->Free(pidl);
300  pMalloc->Release();
301  }
302  }
303  return res;
304 }
