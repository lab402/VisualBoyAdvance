77 void profPut32(char *b, u32 v)
78 {
79  b[0] = v & 255;
80  b[1] = (v >> 8) & 255;
81  b[2] = (v >> 16) & 255;
82  b[3] = (v >> 24) & 255;
83 }

139 void profStartup(u32 lowpc, u32 highpc)
140 {
141  int monsize;
142  char *buffer;
143  int o;
144  profile_segment *newseg = (profile_segment*)calloc(1,sizeof(profile_segment));
145 
146  if (newseg) {
147  newseg->next = first_segment;
148  first_segment = newseg;
149  } else {
150  systemMessage(0, MSG);
151  return;
152  }
153  /*
154  * round lowpc and highpc to multiples of the density we're using
155  * so the rest of the scaling (here and in gprof) stays in ints.
156  */
157  lowpc = ROUNDDOWN(lowpc, HISTFRACTION*sizeof(HISTCOUNTER));
158  newseg->s_lowpc = lowpc;
159  highpc = ROUNDUP(highpc, HISTFRACTION*sizeof(HISTCOUNTER));
160  newseg->s_highpc = highpc;
161  newseg->s_textsize = highpc - lowpc;
162  monsize = (newseg->s_textsize / HISTFRACTION);
163  buffer = (char *)calloc(1, 2*monsize );
164  if ( buffer == NULL ) {
165  systemMessage(0, MSG);
166  return;
167  }
168  newseg->froms = (unsigned short *) calloc(1, 4*newseg->s_textsize / HASHFRACTION );
169  if ( newseg->froms == NULL ) {
170  systemMessage(0, MSG);
171  free(buffer);
172  buffer = NULL;
173  return;
174  }
175  newseg->tolimit = newseg->s_textsize * ARCDENSITY / 100;
176  if ( newseg->tolimit < MINARCS ) {
177  newseg->tolimit = MINARCS;
178  } else if ( newseg->tolimit > 65534 ) {
179  newseg->tolimit = 65534;
180  }
181  newseg->tos = (struct tostruct *) calloc(1, newseg->tolimit * sizeof( struct tostruct ) );
182  if ( newseg->tos == NULL ) {
183  systemMessage(0, MSG);
184 
185  free(buffer);
186  buffer = NULL;
187 
188  free(newseg->froms);
189  newseg->froms = NULL;
190 
191  return;
192  }
193  newseg->tos[0].link = 0;
194  newseg->sbuf = buffer;
195  newseg->ssiz = monsize;
196  if ( monsize <= 0 )
197  return;
198  o = highpc - lowpc;
199  if( monsize < o )
200  newseg->s_scale = (int)(( (float) monsize / o ) * SCALE_1_TO_1);
201  else
202  newseg->s_scale = SCALE_1_TO_1;
203  profControl(1);
204 }

206 void profCleanup()
207 {
208  FILE *fd;
209  int fromindex;
210  int endfrom;
211  u32 frompc;
212  int toindex;
213  struct gmon_hdr ghdr;
214  profile_segment *seg = first_segment;
215 
216  profControl(0);
217  fd = fopen( "gmon.out" , "wb" );
218  if ( fd == NULL ) {
219  systemMessage( 0, "mcount: gmon.out" );
220  return;
221  }
222 
223  memcpy(&ghdr.cookie[0], GMON_MAGIC, 4);
224  profPut32((char *)ghdr.version, GMON_VERSION);
225 
226  if(fwrite(&ghdr, sizeof(ghdr), 1, fd) != 1) {
227  systemMessage(0, "mcount: gmon.out header");
228  fclose(fd);
229  return;
230  }
231 
232  if(hz == 0)
233  hz = 100;
234 
235  while(seg) {
236 
237  hist_num_bins = seg->ssiz;
238 
239  if(profWrite8(fd, GMON_TAG_TIME_HIST) ||
240  profWrite32(fd, (u32)seg->s_lowpc) ||
241  profWrite32(fd, (u32)seg->s_highpc) ||
242  profWrite32(fd, hist_num_bins) ||
243  profWrite32(fd, hz) ||
244  profWrite(fd, hist_dimension, 15) ||
245  profWrite(fd, &hist_dimension_abbrev, 1)) {
246  systemMessage(0, "mcount: gmon.out hist");
247  fclose(fd);
248  return;
249  }
250  u16 *hist_sample = (u16 *)seg->sbuf;
251 
252  u16 count;
253  int i;
254 
255  for(i = 0; i < hist_num_bins; ++i) {
256  profPut16((char *)&count, hist_sample[i]);
257 
258  if(fwrite(&count, sizeof(count), 1, fd) != 1) {
259  systemMessage(0, "mcount: gmon.out sample");
260  fclose(fd);
261  return;
262  }
263  }
264 
265  endfrom = seg->s_textsize / (HASHFRACTION * sizeof(*seg->froms));
266  for ( fromindex = 0 ; fromindex < endfrom ; fromindex++ ) {
267  if ( seg->froms[fromindex] == 0 ) {
268  continue;
269  }
270  frompc = seg->s_lowpc + (fromindex * HASHFRACTION * sizeof(*seg->froms));
271  for (toindex=seg->froms[fromindex]; toindex!=0; toindex=seg->tos[toindex].link) {
272  if(profWrite8(fd, GMON_TAG_CG_ARC) ||
273  profWrite32(fd, (u32)frompc) ||
274  profWrite32(fd, (u32)seg->tos[toindex].selfpc) ||
275  profWrite32(fd, seg->tos[toindex].count)) {
276  systemMessage(0, "mcount: arc");
277  fclose(fd);
278  return;
279  }
280  }
281  }
282  seg = seg->next;
283  }
284  fclose(fd);
285 }

287 void profCount()
288 {
289  register char *selfpc;
290  register unsigned short *frompcindex;
291  register struct tostruct *top;
292  register struct tostruct *prevtop;
293  register long toindex;
294  profile_segment *seg = first_segment;
295 
296  /*
297  * find the return address for mcount,
298  * and the return address for mcount's caller.
299  */
300 
301  /* selfpc = pc pushed by mcount call.
302  This identifies the function that was just entered. */
303  selfpc = (char *) reg[14].I;
304  /* frompcindex = pc in preceding frame.
305  This identifies the caller of the function just entered. */
306  frompcindex = (unsigned short *) reg[12].I;
307  /*
308  * check that we are profiling
309  * and that we aren't recursively invoked.
310  */
311  if (profiling) {
312  goto out;
313  }
314  profiling++;
315  /*
316  * check that frompcindex is a reasonable pc value.
317  * for example: signal catchers get called from the stack,
318  * not from text space. too bad.
319  */
320  while(seg) {
321  u32 index = (long)frompcindex - (long)seg->s_lowpc;
322  if (index <= seg->s_textsize) {
323  frompcindex = (unsigned short *) index;
324  break;
325  }
326  seg = seg->next;
327  }
328  if ((unsigned long) frompcindex > seg->s_textsize) {
329  goto done;
330  }
331  frompcindex =
332  &(seg->froms[((long) frompcindex) / (HASHFRACTION * sizeof(*seg->froms))]);
333  toindex = *frompcindex;
334  if (toindex == 0) {
335  /*
336  * first time traversing this arc
337  */
338  toindex = ++seg->tos[0].link;
339  if (toindex >= seg->tolimit) {
340  goto overflow;
341  }
342  *frompcindex = (unsigned short)toindex;
343  top = &seg->tos[toindex];
344  top->selfpc = selfpc;
345  top->count = 1;
346  top->link = 0;
347  goto done;
348  }
349  top = &seg->tos[toindex];
350  if (top->selfpc == selfpc) {
351  /*
352  * arc at front of chain; usual case.
353  */
354  top->count++;
355  goto done;
356  }
357  /*
358  * have to go looking down chain for it.
359  * top points to what we are looking at,
360  * prevtop points to previous top.
361  * we know it is not at the head of the chain.
362  */
363  for (; /* goto done */; ) {
364  if (top->link == 0) {
365  /*
366  * top is end of the chain and none of the chain
367  * had top->selfpc == selfpc.
368  * so we allocate a new tostruct
369  * and link it to the head of the chain.
370  */
371  toindex = ++seg->tos[0].link;
372  if (toindex >= seg->tolimit) {
373  goto overflow;
374  }
375  top = &seg->tos[toindex];
376  top->selfpc = selfpc;
377  top->count = 1;
378  top->link = *frompcindex;
379  *frompcindex = (unsigned short)toindex;
380  goto done;
381  }
382  /*
383  * otherwise, check the next arc on the chain.
384  */
385  prevtop = top;
386  top = &seg->tos[top->link];
387  if (top->selfpc == selfpc) {
388  /*
389  * there it is.
390  * increment its count
391  * move it to the head of the chain.
392  */
393  top->count++;
394  toindex = prevtop->link;
395  prevtop->link = top->link;
396  top->link = *frompcindex;
397  *frompcindex = (unsigned short)toindex;
398  goto done;
399  }
400 
401  }
402  done:
403  profiling--;
404  /* and fall through */
405  out:
406  return; /* normal return restores saved registers */
407 
408  overflow:
409  profiling++; /* halt further profiling */
410 #define TOLIMIT "mcount: tos overflow\n"
411  systemMessage(0, TOLIMIT);
412  goto out;
413 }
