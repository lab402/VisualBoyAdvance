62  class CRect : public tagRECT
63  //
64  // Wrapper for a RECT structure
65  {
66  public:
67  inline CRect() {}
68  // Initialisation constructor
69  inline explicit CRect( const RECT& rhs ) { Set( rhs.left, rhs.top, rhs.right, rhs.bottom );}
70  inline CRect(int xLeft, int yTop, int xRight, int yBottom) { Set( xLeft, yTop, xRight, yBottom ); }
71  // Get the width of the rectangle
72  inline int Width() const { return right - left; }
73  // Get the height of the rectangle
74  inline int Height() const { return bottom - top; }
75  // overloaded operator so you don't have to do &rc anymore
76  inline operator LPCRECT() const { return this; };
77  inline operator LPRECT() { return this; };
78  // Return the SIZE of the rectangle;
79  inline CSize Size() const { CSize s( Width(), Height() ); return s; }
80  // Return the top left of the rectangle
81  inline POINT TopLeft() const { POINT pt = { left, top }; return pt; }
82  // Return the bottom right of the rectangle
83  inline POINT BottomRight() const { POINT pt = { right, bottom }; return pt; }
84  // Set the rectangles left, top, right and bottom
85  inline void Set( int xLeft, int yTop, int xRight, int yBottom) { top = yTop; bottom = yBottom; right = xRight; left = xLeft; }
86  // Return true if the rectangle contains all zeros
87  inline bool IsEmpty() const { return left == 0 && right == 0 && top == 0 && bottom == 0 ? true : false; }
88  // Zero out our rectangle
89  inline void Empty() { left = right = top = bottom = 0; }
90  // Set the size of the rect but leave the top left position untouched.
91  inline void SetSize( const CSize &size ) { bottom = top + size.cy; right = left + size.cx; }
92  inline void SetSize( const SIZE &size ) { bottom = top + size.cy; right = left + size.cx; }
93  inline void SetSize( int cx, int cy ) { bottom = top + cy; right = left + cx; }
94  // Move the rectangle by an offset
95  inline void Offset( int cx, int cy )
96  {
97  top+=cy;
98  bottom+=cy;
99  right+=cx;
100  left+=cx;
101  }
102  // Inflate the rectangle by the cx and cy, use negative to shrink the rectangle
103  inline void Inflate( int cx, int cy )
104  {
105  top-=cy;
106  bottom+=cy;
107  right+=cx;
108  left-=cx;
109  }
110  // Assignment from a RECT
111  inline CRect &operator = ( const RECT&rhs )
112  {
113  left = rhs.left; top = rhs.top;
114  right = rhs.right; bottom = rhs.bottom;
115  return *this;
116  }
117 
118  // Return true if the point passed is within the rectangle
119  inline bool PtInRect( const POINT &pt ) const { return ( pt.x >= left && pt.x < right && pt.y >=top && pt.y < bottom ); }
120  // Return true if the rectangle passed overlaps this rectangle
121  inline bool Intersect( const RECT &rc ) const { return ( rc.left < right && rc.right > left && rc.top < bottom && rc.bottom > top ); }
122  };
