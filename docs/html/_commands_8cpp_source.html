235 bool winAccelGetID(const char *command, WORD& id)
236 {
237  if(!initialized) {
238  int count = sizeof(winAccelCommands)/sizeof(winAccelCommands[0]);
239 
240  for(int i = 0; i < count; i++) {
241  winAccelStrings.SetAt(winAccelCommands[i].command, winAccelCommands[i].id);
242  }
243  initialized = true;
244  }
245 
246  return winAccelStrings.Lookup(command, id) ? true : false;
247 }
248 
249 void winAccelAddCommands(CAcceleratorManager& mgr)
250 {
251  int count = sizeof(winAccelCommands)/sizeof(winAccelCommands[0]);
252 
253  for(int i = 0; i < count; i++) {
254  if(!mgr.AddCommandAccel(winAccelCommands[i].id, winAccelCommands[i].command, false))
255  mgr.CreateEntry(winAccelCommands[i].id, winAccelCommands[i].command);
256  }

258 }
