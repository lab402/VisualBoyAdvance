110 void PaletteView::save(int which)

114  CString captureBuffer;
115 
116  if(which == 0)
117  captureBuffer = "bg.pal";

119  captureBuffer = "obj.pal";
120 
121  LPCTSTR exts[] = {".pal", ".pal", ".act" };
122 
123  CString filter = theApp.winLoadFilter(IDS_FILTER_PAL);
124  CString title = winResLoadString(IDS_SELECT_PALETTE_NAME);
125  FileDlg dlg(this,
126  captureBuffer,

128  1,
129  "PAL",
130  exts,
131  "",
132  title,
133  true);
134 
135  if(dlg.DoModal() == IDCANCEL) {
136  return;
137  }
138 
139  captureBuffer = dlg.GetPathName();
140 
141  PaletteViewControl *p = NULL;
142 
143  if(which == 0)
144  p = &paletteView;
145  else
146  p = &paletteViewOBJ;
147 
148  switch(dlg.getFilterIndex()) {
149  case 0:
150  case 1:
151  p->saveMSPAL(captureBuffer);
152  break;
153  case 2:
154  p->saveJASCPAL(captureBuffer);

157  p->saveAdobe(captureBuffer);

202 LRESULT PaletteView::OnPalInfo(WPARAM wParam, LPARAM lParam)
203 {
204  u16 color = (u16)wParam;
205  u32 address = (u32)lParam;
206  CString buffer;
207 
208  if(address >= 0x200)
209  address = 0x5000200 + 2*(address & 255);
210  else
211  address = 0x5000000 + 2*(address & 255);
212 
213  buffer.Format("0x%08X", address);
214  GetDlgItem(IDC_ADDRESS)->SetWindowText(buffer);
215 
216  int r = (color & 0x1f);
217  int g = (color & 0x3e0) >> 5;
218  int b = (color & 0x7c00) >> 10;
219 
220  buffer.Format("%d", r);
221  GetDlgItem(IDC_R)->SetWindowText(buffer);
222 
223  buffer.Format("%d", g);
224  GetDlgItem(IDC_G)->SetWindowText(buffer);
225 
226  buffer.Format("%d", b);
227  GetDlgItem(IDC_B)->SetWindowText(buffer);
228 
229  buffer.Format("0x%04X", color);
230  GetDlgItem(IDC_VALUE)->SetWindowText(buffer);
231 
232  colorControl.setColor(color);
233 
234  if(address >= 0x5000200) {
235  paletteView.setSelected(-1);
236  } else
237  paletteViewOBJ.setSelected(-1);
238 
239  return TRUE;
240 }
