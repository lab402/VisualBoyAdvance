205 void GBDisassemble::refresh()
206 {
207  if(gbRom == NULL)
208  return;
209 
210  int h = m_list.GetItemHeight(0);
211  RECT r;
212  m_list.GetClientRect(&r);
213  count = (r.bottom - r.top+1)/h;
214 
215  m_list.ResetContent();
216  if(!emulating || theApp.cartridgeType != 1)
217  return;
218 
219  char buffer[80];
220  u16 addr = address;
221  int i;
222  int sel = -1;
223  for(i = 0; i < count; i++) {
224  if(addr == PC.W)
225  sel = i;
226  addr += gbDis(buffer, addr);
227  m_list.InsertString(-1, buffer);
228  }
229  lastAddress = addr-1;
230  if(sel != -1)
231  m_list.SetCurSel(sel);
232 
233  sprintf(buffer, "%04x", AF.W);
234  GetDlgItem(IDC_R0)->SetWindowText(buffer);
235  sprintf(buffer, "%04x", BC.W);
236  GetDlgItem(IDC_R1)->SetWindowText(buffer);
237  sprintf(buffer, "%04x", DE.W);
238  GetDlgItem(IDC_R2)->SetWindowText(buffer);
239  sprintf(buffer, "%04x", HL.W);
240  GetDlgItem(IDC_R3)->SetWindowText(buffer);
241  sprintf(buffer, "%04x", SP.W);
242  GetDlgItem(IDC_R4)->SetWindowText(buffer);
243  sprintf(buffer, "%04x", PC.W);
244  GetDlgItem(IDC_R5)->SetWindowText(buffer);
245  sprintf(buffer, "%04x", IFF);
246  GetDlgItem(IDC_R6)->SetWindowText(buffer);
247  sprintf(buffer, "%04x", register_LY);
248  GetDlgItem(IDC_LY)->SetWindowText(buffer);
249 
250  m_z = (AF.B.B0 & 0x80) != 0;
251  m_n = (AF.B.B0 & 0x40) != 0;
252  m_h = (AF.B.B0 & 0x20) != 0;
253  m_c = (AF.B.B0 & 0x10) != 0;
254  UpdateData(FALSE);
255 }
