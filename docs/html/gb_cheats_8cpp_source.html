188 void gbAddGsCheat(const char *code, const char *desc)
189 {
190  if(gbCheatNumber > 99) {
191  systemMessage(MSG_MAXIMUM_NUMBER_OF_CHEATS,
192  N_("Maximum number of cheats reached."));
193  return;
194  }
195 
196  if(!gbVerifyGsCode(code)) {
197  systemMessage(MSG_INVALID_GAMESHARK_CODE,
198  N_("Invalid GameShark code: %s"), code);
199  return;
200  }
201 
202  int i = gbCheatNumber;
203 
204  strcpy(gbCheatList[i].cheatCode, code);
205  strcpy(gbCheatList[i].cheatDesc, desc);
206 
207  gbCheatList[i].code = GBCHEAT_HEX_VALUE(code[0]) << 4 |
208  GBCHEAT_HEX_VALUE(code[1]);
209 
210  gbCheatList[i].value = GBCHEAT_HEX_VALUE(code[2]) << 4 |
211  GBCHEAT_HEX_VALUE(code[3]);
212 
213  gbCheatList[i].address = GBCHEAT_HEX_VALUE(code[6]) << 12 |
214  GBCHEAT_HEX_VALUE(code[7]) << 8 |
215  GBCHEAT_HEX_VALUE(code[4]) << 4 |
216  GBCHEAT_HEX_VALUE(code[5]);
217 
218  gbCheatList[i].compare = 0;
219 
220  gbCheatList[i].enabled = true;
221 
222  int gsCode = gbCheatList[i].code;
223 
224  if ((gsCode !=1) && ((gsCode & 0xF0) !=0x80) && ((gsCode & 0xF0) !=0x90) &&
225  ((gsCode & 0xF0) !=0xA0) && ((gsCode) !=0xF0) && ((gsCode) !=0xF1))
226  systemMessage(MSG_WRONG_GAMESHARK_CODE,
227  N_("Wrong GameShark code type : %s"), code);
228  else if (((gsCode & 0xF0) ==0xA0) || ((gsCode) ==0xF0) || ((gsCode) ==0xF1))
229  systemMessage(MSG_UNSUPPORTED_GAMESHARK_CODE,
230  N_("Unsupported GameShark code type : %s"), code);
231 
232  gbCheatNumber++;
233 }

235 bool gbVerifyGgCode(const char *code)
236 {
237  size_t len = strlen(code);
238 
239  if(len != 11 &&
240  len != 7 &&
241  len != 6 &&
242  len != 0)
243  return false;
244 
245  if(len == 0)
246  return true;
247 
248  if(!GBCHEAT_IS_HEX(code[0]))
249  return false;
250  if(!GBCHEAT_IS_HEX(code[1]))
251  return false;
252  if(!GBCHEAT_IS_HEX(code[2]))
253  return false;
254  if(code[3] != '-')
255  return false;
256  if(!GBCHEAT_IS_HEX(code[4]))
257  return false;
258  if(!GBCHEAT_IS_HEX(code[5]))
259  return false;
260  if(!GBCHEAT_IS_HEX(code[6]))
261  return false;
262  if(code[7] != 0) {
263  if(code[7] != '-')
264  return false;
265  if(code[8] != 0) {
266  if(!GBCHEAT_IS_HEX(code[8]))
267  return false;
268  if(!GBCHEAT_IS_HEX(code[9]))
269  return false;
270  if(!GBCHEAT_IS_HEX(code[10]))
271  return false;
272  }
273  }
274 
275  // int replace = (GBCHEAT_HEX_VALUE(code[0]) << 4) +
276  // GBCHEAT_HEX_VALUE(code[1]);
277 
278  int address = (GBCHEAT_HEX_VALUE(code[2]) << 8) +
279  (GBCHEAT_HEX_VALUE(code[4]) << 4) +
280  (GBCHEAT_HEX_VALUE(code[5])) +
281  ((GBCHEAT_HEX_VALUE(code[6]) ^ 0x0f) << 12);
282 
283  if(address >= 0x8000 && address <= 0x9fff)
284  return false;
285 
286  if(address >= 0xc000)
287  return false;
288 
289  if(code[7] == 0 || code[8] == '0')
290  return true;
291 
292  int compare = (GBCHEAT_HEX_VALUE(code[8]) << 4) +
293  (GBCHEAT_HEX_VALUE(code[10]));
294  compare = compare ^ 0xff;
295  compare = (compare >> 2) | ( (compare << 6) & 0xc0);
296  compare ^= 0x45;
297 
298  int cloak = (GBCHEAT_HEX_VALUE(code[8])) ^ (GBCHEAT_HEX_VALUE(code[9]));
299 
300  if(cloak >=1 && cloak <= 7)
301  return false;
302 
303  return true;
304 }

461 void gbCheatWrite(bool reboot)
462 {
463  if(cheatsEnabled)
464  {
465  u16 address = 0;
466 
467  if (gbNextCheat >= gbCheatNumber)
468  gbNextCheat = 0;
469 
470  for(int i = gbNextCheat; i < gbCheatNumber; i++) {
471  if(gbCheatList[i].enabled) {
472  address = gbCheatList[i].address;
473  if ((!reboot) && (address >= 0x8000) && !((address>=0xA000) && (address<0xC000)))
474  { // These codes are executed one per one, at each Vblank
475  switch(gbCheatList[i].code) {
476  case 0x01:
477  gbWriteMemory(address, gbCheatList[i].value);
478  gbNextCheat = i+1;
479  return;
480  case 0x90:
481  case 0x91:
482  case 0x92:
483  case 0x93:
484  case 0x94:
485  case 0x95:
486  case 0x96:
487  case 0x97:
488  case 0x98:
489  case 0x99:
490  case 0x9A:
491  case 0x9B:
492  case 0x9C:
493  case 0x9D:
494  case 0x9E:
495  case 0x9F:
496  int oldbank = gbMemory[0xff70];
497  gbWriteMemory(0xff70, gbCheatList[i].code & 0xf);

501  return;
502  }
503  }

506  switch(gbCheatList[i].code & 0xF0) {
507  case 0x80:
508  gbWriteMemory(0x0000, 0x0A);
